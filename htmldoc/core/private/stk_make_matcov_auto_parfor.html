0031 function K = stk_make_matcov_auto_parfor(model, x, ncores, min_block_size)
0032 
0033 if isstruct(x0), x0 = x0.a; end
0034 
0035 %=== choose the actual block size & number of blocks
0036 B = 0; % number of blocks of x_i's
0037 n = size(x, 1); n_min = ceil(sqrt(min_block_size));
0038 while 1, B = B + 1;
0039     % stop if the blocks are becoming too small
0040     if n/B < n_min, B = max(1,B-1); break; end
0041     % stop if the number of blocks is a multiple of ncores
0042     if mod( B*(B+1)/2, ncores ) == 0, break; end
0043 end
0044 block_size = ceil(n/B); % the last block can be slightly smaller than the others
0045 nb_blocks = B*(B+1)/2;
0046 
0047 %=== prepare blocks
0048 ind = zeros(B,2); % begin/end positions
0049 ind(:,1) = 1 + ((1:B)'-1)*block_size;
0050 ind(:,2) = min( n, ind(:,1)+block_size-1 );
0051 blocks = struct( 'i',cell(1,nb_blocks), 'j',[], 'xi',[], 'xj',[], 'K',[] );
0052 i = 1; j = 0;
0053 for b = 1:nb_blocks;
0054     j = j+1; if(j>i), i=i+1; j=1; end
0055     blocks(b).i = i; blocks(b).xi = x(ind(i,1):ind(i,2),:);
0056     blocks(b).j = j; blocks(b).xj = x(ind(j,1):ind(j,2),:);
0057 end
0058 
0059 %=== process blocks
0060 name = model.covariance_type; param = model.param; % avoids a "parfor" warning
0061 parfor b = 1:nb_blocks,
0062     blocks(b).K = feval( name, param, blocks(b).xi, blocks(b).xj );
0063     % FIXME: avoid computing each term in the covariance matrix twice
0064     % on the diagonal blocks !
0065 end
0066 
0067 %=== piece the whole matrix out from the blocks
0068 K = zeros(n);
0069 for b = 1:nb_blocks,
0070     i = blocks(b).i; j = blocks(b).j;
0071     K( ind(i,1):ind(i,2), ind(j,1):ind(j,2) ) = blocks(b).K;
0072     K( ind(j,1):ind(j,2), ind(i,1):ind(i,2) ) = ( blocks(b).K )';
0073 end
0074 
0075 end % function stk_make_matcov_auto_parfor
