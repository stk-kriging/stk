0032 function K = stk_make_matcov_inter_parfor( x, xco, model, ncores, min_block_size )
0033 
0034 %=== choose the actual block size & number of blocks
0035 B1 = 1; n1 = size(x.a,1);   dB1 = 0;
0036 B2 = 1; n2 = size(xco.a,1); dB2 = 0;
0037 while 1, bs1 = n1/B1; bs2 = n2/B2;
0038     % stop if the blocks are becoming too small
0039     if bs1*bs2 < min_block_size, B1=B1-dB1; B2=B2-dB2; break; end
0040     % stop if the number of blocks is a multiple of ncores
0041     if mod( B1*B2, ncores ) == 0, break; end
0042     % split the largest of both sides
0043     if bs1 > bs2, B1=B1+1; dB1=1; dB2=0;
0044     else B2=B2+1; dB1=0; dB2=1; end
0045 end
0046 bs1 = ceil(bs1); bs2 = ceil(bs2); nb_blocks = B1*B2;
0047 
0048 %=== prepare blocks
0049 ind1 = zeros(B1,2); % begin/end positions in the first dimension
0050 ind1(:,1) = 1 + ((1:B1)'-1)*bs1;
0051 ind1(:,2) = min( n1, ind1(:,1)+bs1-1 );
0052 ind2 = zeros(B2,2); % begin/end positions in the second dimension
0053 ind2(:,1) = 1 + ((1:B2)'-1)*bs2;
0054 ind2(:,2) = min( n2, ind2(:,1)+bs2-1 );
0055 blocks = struct( 'i',cell(1,nb_blocks), 'j',[], 'xi',[], 'xcoj',[], 'K',[] );
0056 for i = 1:B1,
0057     for j = 1:B2,
0058         b = i + (j-1)*B1; % block number
0059         blocks(b).i = i; blocks(b).xi = struct( 'a', x.a(ind1(i,1):ind1(i,2),:) );
0060         blocks(b).j = j; blocks(b).xcoj = struct( 'a', xco.a(ind2(j,1):ind2(j,2),:) );
0061     end
0062 end
0063 
0064 %=== process blocks
0065 name = model.covariance_type; param = model.param; % avoids a "parfor" warning
0066 parfor b = 1:nb_blocks,
0067     blocks(b).K = feval( name, blocks(b).xi, blocks(b).xcoj, param );
0068 end
0069 
0070 %=== piece the whole matrix out from the blocks
0071 K = zeros(n1,n2);
0072 for i = 1:B1,
0073     for j = 1:B2,
0074         b = i + (j-1)*B1; % block number
0075         K( ind1(i,1):ind1(i,2), ind2(j,1):ind2(j,2) ) = blocks(b).K;
0076     end
0077 end
0078
