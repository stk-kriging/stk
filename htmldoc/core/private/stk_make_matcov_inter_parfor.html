0031 function K = stk_make_matcov_inter_parfor(model, x0, x1, ncores, min_block_size)
0032 
0033 if isstruct(x0), x0 = x0.a; end
0034 if isstruct(x1), x1 = x1.a; end
0035 
0036 %=== choose the actual block size & number of blocks
0037 B0 = 1; n0 = size(x0, 1); dB0 = 0;
0038 B1 = 1; n1 = size(x1, 1); dB1 = 0;
0039 while 1, bs0 = n0/B0; bs1 = n1/B1;
0040     % stop if the blocks are becoming too small
0041     if bs0*bs1 < min_block_size, B0=B0-dB0; B1=B1-dB1; break; end
0042     % stop if the number of blocks is a multiple of ncores
0043     if mod( B0*B1, ncores ) == 0, break; end
0044     % split the largest of both sides
0045     if bs0 > bs1, B0=B0+1; dB0=1; dB1=0;
0046     else B1=B1+1; dB0=0; dB1=1; end
0047 end
0048 bs0 = ceil(bs0); bs1 = ceil(bs1); nb_blocks = B0*B1;
0049 
0050 %=== prepare blocks
0051 ind1 = zeros(B0,2); % begin/end positions in the first dimension
0052 ind1(:,1) = 1 + ((1:B0)'-1)*bs0;
0053 ind1(:,2) = min( n0, ind1(:,1)+bs0-1 );
0054 ind2 = zeros(B1,2); % begin/end positions in the second dimension
0055 ind2(:,1) = 1 + ((1:B1)'-1)*bs1;
0056 ind2(:,2) = min( n1, ind2(:,1)+bs1-1 );
0057 blocks = struct( 'i',cell(1,nb_blocks), 'j',[], 'xi',[], 'xcoj',[], 'K',[] );
0058 for i = 1:B0,
0059     for j = 1:B1,
0060         b = i + (j-1)*B0; % block number
0061         blocks(b).i = i; blocks(b).xi = x0(ind1(i,1):ind1(i,2), :);
0062         blocks(b).j = j; blocks(b).xcoj = x1(ind2(j,1):ind2(j,2), :);
0063     end
0064 end
0065 
0066 %=== process blocks
0067 name = model.covariance_type; param = model.param; % avoids a "parfor" warning
0068 parfor b = 1:nb_blocks,
0069     blocks(b).K = feval( name, param, blocks(b).xi, blocks(b).xcoj );
0070 end
0071 
0072 %=== piece the whole matrix out from the blocks
0073 K = zeros(n0,n1);
0074 for i = 1:B0,
0075     for j = 1:B1,
0076         b = i + (j-1)*B0; % block number
0077         K( ind1(i,1):ind1(i,2), ind2(j,1):ind2(j,2) ) = blocks(b).K;
0078     end
0079 end
0080
