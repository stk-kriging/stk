stk_remlqrg

PURPOSE

STK_REMLQRG computes restricted likelihood

SYNOPSIS

function [rl,drl_param,drl_lnv] = stk_remlqrg(xi, yi, model)

DESCRIPTION 

 STK_REMLQRG computes restricted likelihood 

 CALL: [rl,drl_param] = stk_remlqrg(param, xi, yi, model, options)

 STK_REMLQRG computes the restricted likelihood given data and its
 gradient with respect to the parameters of the covariance

0041 PARAMPRIOR = isfield( model, 'prior' );
0042 NOISYOBS   = isfield( model, 'lognoisevariance' );
0043 NOISEPRIOR = isfield( model, 'noiseprior' );
0044 
0045 if ~NOISYOBS, 
0046     if NOISEPRIOR,
0047         error([...
0048             'Having a prior on the noise variance when there is' ...
0049             'no observation noise doesn''t make sense...']);
0050     else
0051         % log(eps) is harmless
0052         model.lognoisevariance = log(eps);
0053     end
0054 end
0055 
0056 n = size(xi.a,1);
0057 
0058 %% compute rl
0059 
0060 [K,P] = stk_make_matcov( xi, model );
0061 q = size(P,2);
0062 
0063 [Q,R_ignored] = qr(P); %#ok<NASGU> %the second argument *must* be here
0064 W = Q(:,(q+1):n);
0065 Wyi = W'*yi.a;
0066 
0067 G = W'*(K*W);
0068 
0069 Ginv = inv(G);
0070 WKWinv_Wyi = Ginv*Wyi; %#ok<MINV>
0071 
0072 [C,p]=chol(G); %#ok<NASGU>
0073 ldetWKW= 2*sum(log(diag(C))); % log(det(G));
0074 
0075 attache= Wyi'*WKWinv_Wyi;
0076 
0077 if PARAMPRIOR
0078     prior = (model.param - model.prior.mean)'*model.prior.invcov*(model.param - model.prior.mean);
0079 else
0080     prior = 0;
0081 end
0082 
0083 if NOISEPRIOR
0084     noiseprior = (model.lognoisevariance - model.noiseprior.mean)^2/model.noiseprior.var;
0085 else
0086     noiseprior = 0;
0087 end
0088 
0089 rl = 1/2*((n-q)*log(2*pi) + ldetWKW + attache + prior + noiseprior);
0090 
0091 %% compute gradient
0092 
0093 if nargout >= 2

0100         WVW = W'*V*W;
0101         drl_param(paramdiff) = 1/2*(sum(sum(Ginv.*WVW)) - WKWinv_Wyi'*WVW*WKWinv_Wyi);
0102     end
0103

0106     end
