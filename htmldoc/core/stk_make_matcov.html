0150 %=== choose the actual block size & number of blocks
0151 B = 0; % number of blocks of x_i's
0152 n = size(x.a,1); n_min = ceil(sqrt(min_block_size));
0153 while 1, B = B + 1;
0154     % stop if the blocks are becoming too small
0155     if n/B < n_min, B = max(1,B-1); break; end
0156     % stop if the number of blocks is a multiple of ncores
0157     if mod( B*(B+1)/2, ncores ) == 0, break; end
0158 end
0159 block_size = ceil(n/B); % the last block can be slightly smaller than the others
0160 nb_blocks = B*(B+1)/2;
0161 
0162 %=== prepare blocks
0163 ind = zeros(B,2); % begin/end positions
0164 ind(:,1) = 1 + ((1:B)'-1)*block_size;
0165 ind(:,2) = min( n, ind(:,1)+block_size-1 );
0166 blocks = struct( 'i',cell(1,nb_blocks), 'j',[], 'xi',[], 'xj',[], 'K',[] );
0167 i = 1; j = 0;
0168 for b = 1:nb_blocks;
0169     j = j+1; if(j>i), i=i+1; j=1; end
0170     blocks(b).i = i; blocks(b).xi = struct( 'a', x.a(ind(i,1):ind(i,2),:) );
0171     blocks(b).j = j; blocks(b).xj = struct( 'a', x.a(ind(j,1):ind(j,2),:) );
0172 end
0173 
0174 %=== process blocks
0175 name = model.covariance_type; param = model.param; % avoids a "parfor" warning
0176 parfor b = 1:nb_blocks,
0177     blocks(b).K = feval( name, blocks(b).xi, blocks(b).xj, param );
0178     % FIXME: avoid computing each term in the covariance matrix twice

0180 end
