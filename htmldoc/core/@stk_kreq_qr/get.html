0029 function value = get (kreq, propname)
0030 
0031 try
0032     
0033     % if it's a field in the underlying structure, don't waste time
0034     value = kreq.(propname);
0035     
0036 catch
0037     
0038     switch propname
0039                     
0040         case 'lambda'
0041             value = kreq.lambda_mu(1:kreq.n, :);
0042             
0043         case 'mu'
0044             value = kreq.lambda_mu((kreq.n + 1):end, :);
0045             
0046         case 'delta_var'
0047             value = dot (kreq.lambda_mu, kreq.RS)';
0048             
0049         case 'log_abs_det_kriging_matrix'
0050             % LS_Q has +/- 1 determinant, so we only need to care about LS_R
0051             value = sum (log (abs (diag (kreq.LS_R))));
0052             
0053         case 'log_det_covariance_matrix_a'
0054             n = kreq.n;
0055             Q11 = kreq.LS_Q(1:n, 1:n);
0056             R11 = kreq.LS_R(1:n, 1:n);
0057             value = log (det (Q11 * R11)); % K = Q11 * R11
0058             % note: this method seems much less accurate than the others,
0059             %       we strongly recommend to avoid using it
0060             
0061         case 'log_det_covariance_matrix_b'
0062             n = kreq.n;
0063             Q11 = kreq.LS_Q(1:n, 1:n);
0064             R11 = kreq.LS_R(1:n, 1:n);
0065             Kchol = chol (Q11 * R11); % K = Q11 * R11
0066             value = 2 * sum (log (diag (Kchol)));
0067             
0068         case 'log_det_covariance_matrix_c'
0069             % note: very efficient (but not the best) and quite simple to understand
0070             n = kreq.n;
0071             Q11 = kreq.LS_Q(1:n, 1:n);
0072             diag_R  = diag (kreq.LS_R);
0073             logdet1 = log (abs (det (Q11)));
0074             logdet2 = sum (log (abs (diag_R(1:n))));
0075             value = logdet1 + logdet2;
0076             
0077         case {'log_det_covariance_matrix', 'log_det_covariance_matrix_d'}
0078             % note: this is the fastest of all five solutions for large matrices
0079             %       (for small matrices, the difference is negligible)... but also
0080             %       the trickiest !
0081             n = kreq.n;
0082             QL = kreq.LS_Q((n + 1):end, :); % lowerpart of LS_Q
0083             diag_R = diag (kreq.LS_R);
0084             logdet1 = sum (log (abs (diag_R)));
0085             T = kreq.LS_R \ (QL');
0086             logdet2 = log (abs (det (T((n + 1):end, :))));
0087             value = logdet1 + logdet2;
0088             
0089         case 'log_det_covariance_matrix_e'
0090             % note: the most time consuming of all five solutions (blame qrdelete),
0091             %       we strongly recommend to avoid using it
0092             n = kreq.n;
0093             r = kreq.r;
0094             Q = kreq.LS_Q;
0095             R = kreq.LS_R;
0096             for i = 0:(r - 1),
0097                 [Q, R] = qrdelete (Q, R, n + r - i, 'row');
0098                 [Q, R] = qrdelete (Q, R, n + r - i, 'col');
0099             end
0100             value = sum (log (abs (diag (R))));
0101             
0102         otherwise
0103             error ('There is no property called ''%s''.', propname);
0104             
0105     end % switch
0106     
0107 end % try_catch
0108 
0109 end % function get
