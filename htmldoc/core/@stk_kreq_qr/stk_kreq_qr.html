0029 function kreq = stk_kreq_qr (model, xi, xt)
0030         
0031 [Kii, Pi] = stk_make_matcov (model, xi);
0032 [n, r] = size (Pi);
0033 
0034 % kriging matrix (left-hand side of the kriging equation)
0035 LS =                              ...
0036     [[ Kii, Pi                 ]; ...
0037     [  Pi', zeros(size(Pi, 2)) ]];
0038         
0039 % orthogonal-triangular decomposition
0040 [Q, R] = qr (LS);
0041 
0042 kreq = struct ('n', n, 'r', r, ...
0043     'LS_Q', Q, 'LS_R', R, 'RS', [], 'lambda_mu', []);
0044 kreq = class (kreq, 'stk_kreq_qr');
0045 
0046 % prepare the right-hand side of the kriging equation
0047 if nargin > 2,
0048     [Kti, Pt] = stk_make_matcov (model, xt, xi);
0049     kreq = stk_set_righthandside (kreq, Kti, Pt);
0050 end
