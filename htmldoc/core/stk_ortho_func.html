0060 function P = stk_ortho_func_( x, order )
0061 
0062 [n,d] = size( x.a );
0063 
0064 switch order
0065     
0066     case -1, % 'simple' kriging
0067         P = [];
0068     
0069     case 0, % 'ordinary' kriging
0070         P = ones(n,1);
0071     
0072     case 1, % linear trend
0073         P = [ ones(n,1) x.a ];
0074     
0075     case 2, % quadratic trend
0076         P = [ ones(n,1) x.a zeros(n,d*(d+1)/2) ];
0077         k = d+2;
0078         for i = 1:d
0079             for j = i:d
0080                 P(:,k) = x.a(:,i) .* x.a(:,j);
0081                 k = k+1;
0082             end
0083         end
0084     
0085     otherwise, % syntax error
