0030 function [zp, kreq] = stk_predict ...
0031     (kreq, zi, xt, display_waitbar, block_size)
0032 
0033 %=== convert zi and check its size
0034 
0035 zi = double(zi);
0036 
0037 ni = size(kreq.xi, 1);  % number of observations
0038 
0039 if ~(isempty(zi) || isequal(size(zi), [ni 1]))
0040     stk_error('zi must have size ni x 1.', 'IncorrectSize');
0041 end
0042 
0043 %=== convert xt and get its size
0044 
0045 xt = double (xt);
0046 
0047 if strcmp (kreq.model.covariance_type, 'stk_discretecov') % use indices
0048     if isempty (xt)
0049         m = size (kreq.model.param.K, 1);
0050         xt = (1:m)';
0051     elseif ~iscolumn (xt)
0052         warning ('STK:stk_predict:IncorrectSize', 'xt should be a column.');
0053         xt = xt(:);
0054     end    
0055 end
0056 
0057 nt = size (xt, 1);
0058 
0059 %=== optional arguments
0060 
0061 if nargin < 4,
0062     display_waitbar = false;
0063 end
0064 
0065 if nargin < 5,
0066     block_size = [];
0067 end
0068 
0069 %=== prepare the output arguments
0070 
0071 zp_v = zeros(nt, 1);
0072 compute_prediction = ~isempty(zi);
0073 
0074 % compute the kriging prediction, or just the variances ?
0075 if compute_prediction,
0076     zp_a = zeros(nt, 1);
0077 else
0078     zp_a = nan(nt, 1);
0079 end
0080     
0081 %=== choose nb_blocks & block_size
0082 
0083 % note: only one block if return_K == true
0084 %       (we need the full set of lambda's and mu's to compute K)
0085 
0086 if isempty(block_size)
0087     MAX_RS_SIZE = 5e6; SIZE_OF_DOUBLE = 8; % in bytes
0088     block_size = ceil(MAX_RS_SIZE / (ni * SIZE_OF_DOUBLE));
0089 end
0090 
0091 % blocks of size approx. block_size
0092 nb_blocks = max(1, ceil(nt / block_size));
0093 
0094 block_size = ceil(nt / nb_blocks);
0095 
0096 % if we want to return a full kreq object in the case where several blocks are
0097 % used, we need to recompose full lambda_mu and RS matrices.
0098 if (nargin > 1) && (nb_blocks > 1)
0099     lambda_mu = zeros(size(kreq.LS_Q, 1), nt);
0100     RS = zeros(size(lambda_mu));
0101 end
0102 
0103 %=== MAIN LOOP (over blocks)
0104 
0105 % TODO: this loop should be parallelized !!!
0106 
0107 for block_num = 1:nb_blocks
0108     
0109     % compute the indices for the current block
0110     idx_beg = 1 + block_size * (block_num - 1);
0111     idx_end = min(nt, idx_beg + block_size - 1);
0112     idx = idx_beg:idx_end;
0113         
0114     % solve the kriging equation for the current block
0115     kreq = set (kreq, 'xt', xt(idx, :));
0116 
0117     % compute the kriging mean
0118     if compute_prediction,
0119         zp_a(idx) = (kreq.lambda_mu(1:ni, :))' * zi;
0120     end
0121 
0122     if (nargin > 1) && (nb_blocks > 1)
0123         lambda_mu(:, idx) = kreq.lambda_mu;
0124         RS(:, idx) = kreq.RS;
0125     end
0126     
0127     % compute kriging variances (this does NOT include the noise variance)
0128     zp_v(idx) = stk_posterior_matcov(kreq, 1:length(idx), 1:length(idx), true);
0129     
0130     % note: the following modification computes prediction variances for noisy
0131     % variance, i.e., including the noise variance also
0132     % zp_v(idx) = stk_posterior_matcov(kreq, 1:nt, [], true);
0133     
0134     b = (zp_v < 0);
0135     if any(b),
0136         zp_v(b) = 0.0;
0137         warning('STK:stk_predict:NegativeVariancesSetToZero', sprintf( ...
0138             ['Correcting numerical inaccuracies in kriging variance.\n' ...
0139             '(%d negative variances have been set to zero)'], sum(b)));
0140     end
0141     
0142     if display_waitbar,
0143         waitbar(idx_end/nt, hwb, sprintf( ...
0144             'In stk\\_predict(): %d/%d predictions completed',idx_end,nt));
0145     end
0146 end
0147 
0148 if display_waitbar,
0149     close(hwb);
0150 end
0151 
0152 %=== Prepare outputs
0153 
0154 zp = stk_dataframe([zp_a zp_v], {'mean' 'var'});
0155 
0156 if (nargin > 1) && (nb_blocks > 1)
0157     kreq.lambda_mu = lambda_mu;
0158     kreq.RS = RS;
0159 end
0160 
0161 end
