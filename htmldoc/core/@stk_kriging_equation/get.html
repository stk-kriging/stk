0029 function value = get (kreq, propname)
0030 
0031 switch propname
0032     
0033     case 'log_abs_det_kriging_matrix'
0034         % LS_Q has +/- 1 determinant, so we only need to care about LS_R
0035         value = sum (log (abs (diag (kreq.LS_R))));
0036                  
0037     case 'log_det_covariance_matrix_a'
0038         n = size (kreq.xi, 1);
0039         Q11 = kreq.LS_Q(1:n, 1:n);
0040         R11 = kreq.LS_R(1:n, 1:n);
0041         value = log (det (Q11 * R11)); % K = Q11 * R11
0042         % note: this method seems much less accurate than the others,
0043         %       we strongly recommend to avoid using it
0044         
0045     case 'log_det_covariance_matrix_b'
0046         n = size (kreq.xi, 1);
0047         Q11 = kreq.LS_Q(1:n, 1:n);
0048         R11 = kreq.LS_R(1:n, 1:n);
0049         Kchol = chol (Q11 * R11); % K = Q11 * R11
0050         value = 2 * sum (log (diag (Kchol)));
0051         
0052     case 'log_det_covariance_matrix_c'
0053         % note: very efficient (but not the best) and quite simple to understand
0054         n = size(kreq.xi, 1);
0055         Q11 = kreq.LS_Q(1:n, 1:n);
0056         diag_R  = diag (kreq.LS_R);
0057         logdet1 = log (abs (det (Q11)));
0058         logdet2 = sum (log (abs (diag_R(1:n))));
0059         value = logdet1 + logdet2;
0060         
0061     case {'log_det_covariance_matrix', 'log_det_covariance_matrix_d'}
0062         % note: this is the fastest of all five solutions for large matrices
0063         %       (for small matrices, the difference is negligible)... but also
0064         %       the trickiest !
0065         n = size (kreq.xi, 1);
0066         QL = kreq.LS_Q((n + 1):end, :); % lowerpart of LS_Q
0067         diag_R = diag (kreq.LS_R);
0068         logdet1 = sum (log (abs (diag_R)));
0069         T = kreq.LS_R \ (QL');
0070         logdet2 = log (abs (det (T((n + 1):end, :))));
0071         value = logdet1 + logdet2;
0072         
0073     case 'log_det_covariance_matrix_e'
0074         % note: the most time consuming of all five solutions (blame qrdelete),
0075         %       we strongly recommend to avoid using it
0076         n = size (kreq.xi, 1);
0077         Q = kreq.LS_Q;
0078         R = kreq.LS_R;
0079         r = size (Q, 1) - n;
0080         for i = 0:(r - 1),
0081             [Q, R] = qrdelete (Q, R, n + r - i, 'row');
0082             [Q, R] = qrdelete (Q, R, n + r - i, 'col');
0083         end
0084         value = sum (log (abs (diag (R))));
0085         
0086     otherwise
0087         try
0088             value = kreq.(propname);
0089         catch
0090             error ('There is no property called ''%s''.', propname);
0091         end
0092         
0093 end % switch
0094 
0095 end % function get
