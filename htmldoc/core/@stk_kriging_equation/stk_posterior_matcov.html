0030 function Kpost = stk_posterior_matcov(kreq, idx1, idx2, pairwise)
0031 
0032 if nargin < 4,
0033     pairwise = false;
0034 end
0035 
0036 xt1 = kreq.xt(idx1, :);
0037 
0038 if ~isempty(idx2)
0039     xt2 = kreq.xt(idx2, :);
0040 end
0041 
0042 if isempty(idx2) % Compute K(xt1, xt1)
0043     
0044     % WARNING: this form adds the variance of the noise on the diagonal
0045     
0046     if pairwise    
0047         % Compute a vector of posterior variances
0048         % (this is the variance of future noisy observations)
0049         Kpost = stk_make_matcov(kreq.model, xt1, [], true) ...
0050             - dot(kreq.lambda_mu, kreq.RS)';
0051     else
0052         % Compute a posterior covariance matrix
0053         K0 = stk_make_matcov(kreq.model, xt1, [], false);
0054         Ka = kreq.lambda_mu(:, idx1)' * kreq.RS(:, idx1);
0055         Kpost = K0 - .5 * (Ka + Ka');
0056     end
0057     
0058 else % Compute K(xt1, xt2)
0059     
0060     % WARNING: this form does NOT add the variance of the noise even if xt1 and
0061     % xt2 are equal (or have some elements in common)
0062 
0063     K0 = stk_make_matcov(kreq.model, xt1, xt2, pairwise);
0064 
0065     if pairwise % Compute a vector of posterior covariances
0066         
0067         K1 = dot(kreq.lambda_mu(:, idx1), kreq.RS(:, idx2))';
0068         
0069         if ~isequal(idx1, idx2)
0070             K1 = .5 * (K1 + dot(kreq.lambda_mu(:, idx2), kreq.RS(:, idx1))');
0071         end       
0072         
0073     else % Compute a posterior covariance matrix
0074         
0075         K1 = kreq.lambda_mu(:, idx1)' * kreq.RS(:, idx2);
0076         
0077         if isequal(idx1, idx2)
0078             K1 = .5 * (K1 + K1');
0079         else
0080             K1 = .5 * (K1 + kreq.RS(:, idx1)' * kreq.lambda_mu(:, idx2));
0081         end
0082 
0083     end
0084     
0085     Kpost = K0 - K1;
0086     
0087 end
0088 
0089 end % function stk_posterior_matcov
