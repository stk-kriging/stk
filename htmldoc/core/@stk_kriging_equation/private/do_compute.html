0030 function kreq = do_compute (kreq)
0031 
0032 if  ~ isempty (kreq.xi)
0033 
0034     %--- step #1: QR factorization of the kriging equation LHS -----------------
0035     
0036     if isempty (kreq.LS_Q)
0037         
0038         [Kii, Pi] = stk_make_matcov(kreq.model, kreq.xi);
0039         
0040         LS =                              ...
0041             [[ Kii, Pi                 ]; ...
0042             [  Pi', zeros(size(Pi, 2)) ]];
0043         
0044         % orthogonal-triangular decomposition
0045         [kreq.LS_Q, kreq.LS_R] = qr (LS);
0046 
0047     end
0048     
0049     %--- step #2: compute kriging weights --------------------------------------
0050     
0051     if (~ isempty (kreq.xt)) && (isempty (kreq.RS))
0052         
0053         % prepare the right-hand side of the kriging equation
0054         [Kti, Pt] = stk_make_matcov (kreq.model, kreq.xt, kreq.xi);
0055         kreq.RS = [Kti Pt]';
0056         
0057         % Solve the kriging equation to get the extended
0058         % kriging weights vector (weights + Lagrange multipliers)
0059         kreq.lambda_mu = linsolve (kreq, kreq.RS);
0060         
0061     end
0062     
0063 end
