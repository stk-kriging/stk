stk_param_init

PURPOSE

STK_PARAM_INIT provides a starting point for stk_param_estim().

SYNOPSIS

function [param, lnv] = stk_param_init(model, xi, yi, box, noisy)

DESCRIPTION 

 STK_PARAM_INIT provides a starting point for stk_param_estim().

0029 function [param, lnv] = stk_param_init(model, xi, yi, box, noisy)
0030 if nargin > 5,
0031    stk_error ('Too many input arguments.', 'TooManyInputArgs');
0032 end
0033 
0034 %--- first, default values for arguments 'box' and 'noisy' ------------------------------
0035 
0036 if (nargin < 4) || isempty(box),
0037     box = [min(xi); max(xi)];
0038 end
0039 
0040 if nargin < 5,
0041     noisy = false;
0042 end
0043 
0044 %--- lognoisevariance ? -----------------------------------------------------------------
0045 
0046 if isfield(model, 'lognoisevariance') && ~isempty(model.lognoisevariance)
0047     warning(sprintf('Ignoring current value of lognoisevariance (%.3e)', ...
0048         model.lognoisevariance), 'IgnoringLogNoiseVariance');
0049 end
0050 
0051 %--- then, each type of covariance is dealt with specifically ---------------------------
0052 
0053 switch model.covariance_type
0054     
0055     case 'stk_materncov_iso'
0056         nu = 5/2 * size(xi, 2);
0057         [param, lnv] = paraminit_(xi, yi, box, nu, model.order, noisy);
0058         
0059     case 'stk_materncov_aniso'
0060         nu = 5/2 * size(xi, 2);
0061         xi = stk_normalize(xi, box);
0062         [param, lnv] = paraminit_(xi, yi, [], nu, model.order, noisy);
0063         param = [param(1:2); param(3) - log(diff(box, [], 1))'];
0064         
0065     case 'stk_materncov32_iso'
0066         [param, lnv] = paraminit_(xi, yi, box, 3/2, model.order, noisy);
0067         param = [param(1); param(3)];
0068         
0069     case 'stk_materncov32_aniso'
0070         xi = stk_normalize(xi, box);
0071         [param, lnv] = paraminit_(xi, yi, [], 3/2, model.order, noisy);
0072         param = [param(1); param(3) - log(diff(box, [], 1))'];
0073         
0074     case 'stk_materncov52_iso'
0075         [param, lnv] = paraminit_(xi, yi, box, 5/2, model.order, noisy);
0076         param = [param(1); param(3)];
0077         
0078     case 'stk_materncov52_aniso'
0079         xi = stk_normalize(xi, box);
0080         [param, lnv] = paraminit_(xi, yi, [], 5/2, model.order, noisy);
0081         param = [param(1); param(3) - log(diff(box, [], 1))'];
0082         
0083     otherwise
0084         errmsg = 'Unsupported covariance type.';
0085         stk_error(errmsg, 'IncorrectArgument');

0091 function [param, lnv] = paraminit_(xi, yi, box, nu, order, noisy)
0092 
0093 [ni d] = size(xi);
0094 
0095 model = stk_model('stk_materncov_iso');
0096 model.order = order;
0097 
0098 % list of possible values for the ratio eta = sigma2_noise / sigma2
0099 if noisy,
0100     eta_list = 10 .^ (-6:3:0);
0101 else
0102     eta_list = 1e-10;
0103 end
0104 
0105 % list of possible values for the range parameter
0106 if isempty(box)
0107     % assume box = repmat([0; 1], 1, d)
0108     box_diameter = sqrt(d);
0109 else
0110     box_diameter = sqrt(sum(diff(box) .^ 2));
0111 end
0112 rho_max  = 2 * box_diameter;
0113 rho_min  = box_diameter / 50;
0114 rho_list = logspace(log10(rho_min), log10(rho_max), 5);
0115 
0116 % try all possible combinations
0117 eta_best    = NaN;
0118 rho_best    = NaN;
0119 sigma2_best = NaN;
0120 aLL_best    = +Inf;
0121 for eta = eta_list
0122     for rho = rho_list
0123         fprintf('[stk_param_init] eta = %.3e, rho = %.3e...\n', eta, rho);
0124         % first use sigma2 = 1
0125         model.param = log([1.0; nu; 1/rho]);
0126         model.lognoisevariance = log(eta);
0127         [K, P] = stk_make_matcov(model, xi);
0128         % estimate sigma2
0129         % (TODO: use Cholesky ?)
0130         yi_ = double(yi);
0131         beta = (P' * (K \ P)) \ (P' * yi_);
0132         zi = yi_ - P * beta;
0133         sigma2 = 1 / (ni - length(beta)) * zi' * (K \ zi);
0134         % now compute the antilog-likelihood
0135         if sigma2 > 0
0136             model.param(1) = log(sigma2);
0137             model.lognoisevariance = log(eta * sigma2);
0138             aLL = stk_param_relik (model, xi, yi);
0139             if ~isnan(aLL) && (aLL < aLL_best)
0140                 eta_best    = eta;
0141                 rho_best    = rho;
0142                 aLL_best    = aLL;
0143                 sigma2_best = sigma2;
0144             end
0145             %fprintf('eta = %.2e,  rho = %.2e,  aLL=%.2e\n', eta, rho, aLL);
0146         end
0147     end
0148 end
0149 
0150 if isinf(aLL_best)
0151     errmsg = 'Couldn''t find reasonable parameter values... ?!?';
0152     stk_error(errmsg, 'AlgorithmFailure');
0153 end
0154 
0155 param = log([sigma2_best; nu; 1/rho_best]);
0156 lnv = log(eta_best * sigma2_best);
0157 
0158 end % function paraminit_
