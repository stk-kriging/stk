stk_param_init

PURPOSE

STK_PARAM_INIT provides a starting point for stk_param_estim().

SYNOPSIS

function [param, lnv] = stk_param_init (model, xi, zi, box, noisy)

DESCRIPTION 

 STK_PARAM_INIT provides a starting point for stk_param_estim().

CROSS-REFERENCE INFORMATION

This function calls:
  • stk_param_relik STK_REMLQRG computes the restricted likelihood of a model given data.

0029 function [param, lnv] = stk_param_init (model, xi, zi, box, noisy)
0030 
0031 if nargin > 5,
0032    stk_error ('Too many input arguments.', 'TooManyInputArgs');
0033 end
0034 
0035 if ~ isequal (size (zi), [size(xi, 1) 1]),
0036     errmsg = 'zi should be a column, with the same number of rows as xi.';
0037     stk_error (errmsg, 'IncorrectSize');
0038 end
0039 
0040 %--- first, default values for arguments 'box' and 'noisy' ---------------------
0041 
0042 if (nargin < 4) || isempty(box),
0043     box = [min(xi); max(xi)];
0044 end
0045 
0046 if nargin < 5,
0047     noisy = false;
0048 end
0049 
0050 %--- lognoisevariance ? --------------------------------------------------------
0051 
0052 if isfield (model, 'lognoisevariance') && ~ isempty (model.lognoisevariance)
0053     warning (sprintf ('Ignoring current value of lognoisevariance (%.3e)', ...
0054         model.lognoisevariance), 'IgnoringLogNoiseVariance');
0055 end

0057 %--- then, each type of covariance is dealt with specifically ------------------
0058 
0059 switch model.covariance_type
0060     
0061     case 'stk_materncov_iso'
0062         nu = 5/2 * size (xi, 2);
0063         [param, lnv] = paraminit_ (xi, zi, box, nu, model.order, noisy);
0064         
0065     case 'stk_materncov_aniso'
0066         nu = 5/2 * size (xi, 2);
0067         xi = stk_normalize (xi, box);
0068         [param, lnv] = paraminit_ (xi, zi, [], nu, model.order, noisy);
0069         param = [param(1:2); param(3) - log(diff(box, [], 1))'];
0070         
0071     case 'stk_materncov32_iso'
0072         [param, lnv] = paraminit_ (xi, zi, box, 3/2, model.order, noisy);
0073         param = [param(1); param(3)];
0074         
0075     case 'stk_materncov32_aniso'
0076         xi = stk_normalize (xi, box);
0077         [param, lnv] = paraminit_ (xi, zi, [], 3/2, model.order, noisy);

0079         
0080     case 'stk_materncov52_iso'

0097 function [param, lnv] = paraminit_ (xi, zi, box, nu, order, noisy)
0098 
0099 [ni d] = size (xi);
0100 
0101 model = stk_model ('stk_materncov_iso');
0102 model.order = order;
0103 
0104 % list of possible values for the ratio eta = sigma2_noise / sigma2
0105 if noisy,
0106     eta_list = 10 .^ (-6:3:0);
0107 else
0108     eta_list = 1e-10;
0109 end
0110 
0111 % list of possible values for the range parameter
0112 if isempty (box)
0113     % assume box = repmat([0; 1], 1, d)
0114     box_diameter = sqrt (d);
0115 else
0116     box_diameter = sqrt (sum (diff (box) .^ 2));
0117 end
0118 rho_max  = 2 * box_diameter;
0119 rho_min  = box_diameter / 50;
0120 rho_list = logspace (log10 (rho_min), log10 (rho_max), 5);
0121 
0122 % try all possible combinations
0123 eta_best    = NaN;
0124 rho_best    = NaN;
0125 sigma2_best = NaN;
0126 aLL_best    = +Inf;
0127 for eta = eta_list
0128     for rho = rho_list
0129         fprintf ('[stk_param_init] eta = %.3e, rho = %.3e...\n', eta, rho);
0130         % first use sigma2 = 1.0
0131         model.param = log ([1.0; nu; 1/rho]);
0132         model.lognoisevariance = log (eta);
0133         [K, P] = stk_make_matcov (model, xi);
0134         % estimate sigma2
0135         zi = double (zi);  L = chol (K, 'lower');  W = L \ P;
0136         beta = (W' * W) \ (W' * (L \ zi));
0137         sigma2 = 1 / (ni - length (beta)) * sum ((L \ (zi - P * beta)) .^ 2);
0138         % now compute the antilog-likelihood
0139         if sigma2 > 0
0140             model.param(1) = log (sigma2);
0141             model.lognoisevariance = log  (eta * sigma2);
0142             aLL = stk_param_relik (model, xi, zi);
0143             if ~isnan(aLL) && (aLL < aLL_best)
0144                 eta_best    = eta;
0145                 rho_best    = rho;
0146                 aLL_best    = aLL;
0147                 sigma2_best = sigma2;
0148             end
0149             %fprintf ('eta = %.2e,  rho = %.2e,  aLL=%.2e\n', eta, rho, aLL);
0150         end
0151     end
0152 end
0153 
0154 if isinf (aLL_best)
0155     errmsg = 'Couldn''t find reasonable parameter values... ?!?';
0156     stk_error (errmsg, 'AlgorithmFailure');
0157 end
0158 
0159 param = log ([sigma2_best; nu; 1/rho_best]);
0160 lnv = log (eta_best * sigma2_best);
0161 
0162 end % function paraminit_
