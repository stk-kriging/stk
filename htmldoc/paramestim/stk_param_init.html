0084 function [param, lnv] = paraminit_(xi, yi, box, nu, order, noisy)
0085 
0086 model = stk_model('stk_materncov_iso');
0087 model.order = order;
0088 
0089 % normalized factors
0090 xi = stk_normalize(xi, box);
0091 
0092 % list of possible values for the ratio eta = sigma2_noise / sigma2
0093 if noisy,
0094     eta_list = 10 .^ (-6:3:0);
0095 else
0096     eta_list = 1e-10;
0097 end
0098 
0099 % list of possible values for the range parameter
0100 rho_list = sqrt(size(box, 2)) * 10 .^ (-2:.5:0);
0101 
0102 % try all possible combinations
0103 eta_best    = NaN;
0104 rho_best    = NaN;
0105 sigma2_best = NaN;
0106 aLL_best    = +Inf;
0107 for eta = eta_list
0108     for rho = rho_list
0109         % first use sigma2 = 1
0110         model.param = log([1.0; nu; 1/rho]);
0111         model.lognoisevariance = log(eta);
0112         [K, P] = stk_make_matcov(model, xi);
0113         % estimate sigma2
0114         % (TODO: use Cholesky ?)
0115         beta = (P' * (K \ P)) \ (P' * yi.a);
0116         zi = yi.a - P * beta;
0117         sigma2 = 1 / (size(xi.a, 1) - length(beta)) * zi' * (K \ zi);
0118         % now compute the antilog-likelihood
0119         if sigma2 > 0
0120             model.param(1) = log(sigma2);
0121             model.lognoisevariance = log(eta * sigma2);
0122             aLL = stk_remlqrg(model, xi, yi);
0123             if ~isnan(aLL) && (aLL < aLL_best)
0124                 eta_best    = eta;
0125                 rho_best    = rho;
0126                 aLL_best    = aLL;
0127                 sigma2_best = sigma2;
0128             end
0129             %fprintf('eta = %.2e,  rho = %.2e,  aLL=%.2e\n', eta, rho, aLL);
0130         end
0131     end
0132 end
0133 
0134 if isinf(aLL_best)
0135     errmsg = 'Couldn''t find reasonable parameter values... ?!?';
0136     stk_error(errmsg, 'AlgorithmFailure');
0137 end
0138 
0139 param = log([sigma2_best; nu; 1/rho_best]);
0140 lnv = log(eta_best * sigma2_best);
0141 
0142 end % function paraminit_
