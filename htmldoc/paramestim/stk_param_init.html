stk_param_init

PURPOSE

STK_PARAM_INIT provides a starting point for stk_param_estim().

SYNOPSIS

function [param, lnv] = stk_param_init(model, xi, yi, box, noisy)

DESCRIPTION 

 STK_PARAM_INIT provides a starting point for stk_param_estim().

0029 function [param, lnv] = stk_param_init(model, xi, yi, box, noisy)
0030 
0031 if nargin > 5,
0032    stk_error ('Too many input arguments.', 'TooManyInputArgs');
0033 end
0034 
0035 %--- first, default values for arguments 'box' and 'noisy' ------------------------------
0036 
0037 if (nargin < 4) || isempty(box),
0038     box = [min(xi); max(xi)];
0039 end
0040 
0041 if nargin < 5,
0042     noisy = false;
0043 end
0044 
0045 %--- lognoisevariance ? -----------------------------------------------------------------
0046 
0047 if isfield(model, 'lognoisevariance') && ~isempty(model.lognoisevariance)
0048     warning(sprintf('Ignoring current value of lognoisevariance (%.3e)', ...
0049         model.lognoisevariance), 'IgnoringLogNoiseVariance');
0050 end
0051 
0052 %--- then, each type of covariance is dealt with specifically ---------------------------
0053 
0054 switch model.covariance_type
0055     
0056     case 'stk_materncov_iso'
0057         nu = 5/2 * size(xi, 2);
0058         [param, lnv] = paraminit_(xi, yi, box, nu, model.order, noisy);
0059         
0060     case 'stk_materncov_aniso'
0061         nu = 5/2 * size(xi, 2);
0062         xi = stk_normalize(xi, box);
0063         [param, lnv] = paraminit_(xi, yi, [], nu, model.order, noisy);
0064         param = [param(1:2); param(3) - log(diff(box, [], 1))'];
0065         
0066     case 'stk_materncov32_iso'
0067         [param, lnv] = paraminit_(xi, yi, box, 3/2, model.order, noisy);
0068         param = [param(1); param(3)];
0069         
0070     case 'stk_materncov32_aniso'
0071         xi = stk_normalize(xi, box);
0072         [param, lnv] = paraminit_(xi, yi, [], 3/2, model.order, noisy);
0073         param = [param(1); param(3) - log(diff(box, [], 1))'];
0074         
0075     case 'stk_materncov52_iso'
0076         [param, lnv] = paraminit_(xi, yi, box, 5/2, model.order, noisy);
0077         param = [param(1); param(3)];
0078         
0079     case 'stk_materncov52_aniso'
0080         xi = stk_normalize(xi, box);
0081         [param, lnv] = paraminit_(xi, yi, [], 5/2, model.order, noisy);
0082         param = [param(1); param(3) - log(diff(box, [], 1))'];
0083         
0084     otherwise
0085         errmsg = 'Unsupported covariance type.';
0086         stk_error(errmsg, 'IncorrectArgument');

0092 function [param, lnv] = paraminit_(xi, yi, box, nu, order, noisy)
0093 
0094 [ni d] = size(xi);
0095 
0096 model = stk_model('stk_materncov_iso');
0097 model.order = order;
0098 
0099 % list of possible values for the ratio eta = sigma2_noise / sigma2
0100 if noisy,
0101     eta_list = 10 .^ (-6:3:0);
0102 else
0103     eta_list = 1e-10;
0104 end
0105 
0106 % list of possible values for the range parameter
0107 if isempty(box)
0108     % assume box = repmat([0; 1], 1, d)
0109     box_diameter = sqrt(d);
0110 else
0111     box_diameter = sqrt(sum(diff(box) .^ 2));
0112 end
0113 rho_max  = 2 * box_diameter;
0114 rho_min  = box_diameter / 50;
0115 rho_list = logspace(log10(rho_min), log10(rho_max), 5);
0116 
0117 % try all possible combinations
0118 eta_best    = NaN;
0119 rho_best    = NaN;
0120 sigma2_best = NaN;
0121 aLL_best    = +Inf;
0122 for eta = eta_list
0123     for rho = rho_list
0124         fprintf('[stk_param_init] eta = %.3e, rho = %.3e...\n', eta, rho);
0125         % first use sigma2 = 1
0126         model.param = log([1.0; nu; 1/rho]);
0127         model.lognoisevariance = log(eta);
0128         [K, P] = stk_make_matcov(model, xi);
0129         % estimate sigma2
0130         % (TODO: use Cholesky ?)
0131         yi_ = double(yi);
0132         beta = (P' * (K \ P)) \ (P' * yi_);
0133         zi = yi_ - P * beta;
0134         sigma2 = 1 / (ni - length(beta)) * zi' * (K \ zi);
0135         % now compute the antilog-likelihood
0136         if sigma2 > 0
0137             model.param(1) = log(sigma2);
0138             model.lognoisevariance = log(eta * sigma2);
0139             aLL = stk_param_relik (model, xi, yi);
0140             if ~isnan(aLL) && (aLL < aLL_best)
0141                 eta_best    = eta;
0142                 rho_best    = rho;
0143                 aLL_best    = aLL;
0144                 sigma2_best = sigma2;
0145             end
0146             %fprintf('eta = %.2e,  rho = %.2e,  aLL=%.2e\n', eta, rho, aLL);
0147         end
0148     end
0149 end
0150 
0151 if isinf(aLL_best)
0152     errmsg = 'Couldn''t find reasonable parameter values... ?!?';
0153     stk_error(errmsg, 'AlgorithmFailure');
0154 end
0155 
0156 param = log([sigma2_best; nu; 1/rho_best]);
0157 lnv = log(eta_best * sigma2_best);
0158 
0159 end % function paraminit_
