0039 % final size of the desired DoE
0040 NB_ITER = 10;
0041 
0042 % plot styles
0043 STYLE_CURRENT = {'bo', 'MarkerFaceColor', 0.7 * ones(1, 3)};
0044 STYLE_NEXT = {'ro', 'MarkerFaceColor', 'y'};

0047 %% Sequential design
0048 %
0049 % This section can be run several times, to produce a bigger DoE !
0050 %
0051 
0052 for i = 1:NB_ITER,
0053     % compute the current fill distance & the point where the maximum is attained
0054     [fd, next_x] = stk_filldist(x, BOX);
0055     % plot
0056     figure(1); cla; plot(x(:, 1), x(:, 2), STYLE_CURRENT{:});
0057     hold on; plot(next_x(:, 1), next_x(:, 2), STYLE_NEXT{:});
0058     title(sprintf('n = %d,  fd = %.2e\n', size(x, 1), fd));
0059     drawnow; pause(0.5);
0060     % enrich the DoE
0061     x = vertcat(x, next_x);
0062 end
