0035 %% Define a 1d test function
0036 
0037 f = @(x)( -(0.7*x+sin(5*x+1)+0.1*sin(10*x)) );  % define a 1D test function
0038 DIM = 1;                                        % dimension of the factor space
0039 BOX = [-1.0; 1.0];                              % factor space
0040 
0041 NT = 400; % nb of points in the grid
0042 xt = stk_sampling_regulargrid(NT, DIM, BOX);
0043 zt = stk_feval(f, xt);
0044 
0045 
0046 %% Generate observations
0047 %
0048 % The objective is to construct an approximation of f and to simulate
0049 % conditioned sample paths from NI observations. The observation locations
0050 % are chosen as a subset of xt.
0051 %
0052 
0053 NI = 6;                               % nb of evaluations that will be used
0054 xi_ind  = [1 20 90 200 300 350];      %
0055 xi = xt(xi_ind, 1);
0056 zi = stk_feval(f, xi);                % evaluation results
0057 
0058 
0059 %% Specification of the model
0060 %
0061 % We choose a Matern covariance with "fixed parameters" (in other
0062 % words, the parameters of the covariance function are provided by the user
0063 % rather than estimated from data).
0064 %
0065 
0066 % The following line defines a model with a constant but unknown mean
0067 % (ordinary kriging) and a Matern covariance function. (Some default
0068 % parameters are also set, but we override them below.)
0069 model = stk_model('stk_materncov_iso');
0070 
0071 % Parameters for the Matern covariance
0072 % ("help stk_materncov_iso" for more information)
0073 SIGMA2 = 1.0;  % variance parameter
0074 NU     = 4.0;  % regularity parameter
0075 RHO1   = 0.4;  % scale (range) parameter
0076 model.param = log([SIGMA2; NU; 1/RHO1]);
0077 
0078 
0079 %% Carry out the kriging prediction and generate conditional sample paths
0080 %
0081 
0082 % Carry out the kriging prediction at points xt
0083 [zp, lambda] = stk_predict(model, xi, zi, xt);
0084 
0085 % Generate (unconditional) sample paths according to the model
0086 NB_PATHS = 10;
0087 zsim = stk_generate_samplepaths(model, xt, NB_PATHS);
0088 
0089 % Condition sample paths on the observations
0090 zsimc = stk_conditioning(lambda, zi, zsim, xi_ind);
0091 
0092 % Display the result
0093 stk_plot1dsim(xi, zi, xt, zt, zp, zsimc);
0094 t = 'Kriging prediction and conditional sample paths';
0095 set(gcf, 'Name', t); title(t);
