0039 %% CHOICE OF A TWO-DIMENSIONAL TEST FUNCTION
0040 
0041 CASENUM = 1;
0042 
0043 switch CASENUM
0044

0046         f = @stk_testfun_braninhoo;

0049         NI = 20;

0051     case 2,  % another test function
0052         f_ = inline(['exp(1.8*(x1+x2)) + 3*x1 + 6*x2.^2' ...

0062 %% COMPUTE AND VISUALIZE THE FUNCTION ON A 80 x 80 REGULAR GRID
0063 
0064 % Size of the regular grid
0065 NT = 80^2;
0066 
0067 % The function stk_sampling_regulargrid() does the job of creating the grid
0068 xt = stk_sampling_regulargrid(NT, DIM, BOX);
0069 
0070 % Compute the corresponding responses
0071 zt = stk_feval(f, xt);
0072 
0073 % Since xt is a regular grid, we can do a contour plot
0074 figure; h1 = subplot(2, 2, 1); stk_plot2d(@contour, xt, f, CONTOUR_LINES);
0075 axis(BOX(:)); title('function to be approximated');

0077 
0078 %% CHOOSE A KRIGING (GAUSSIAN PROCESS) MODEL
0079 
0080 % We start with a generic (anisotropic) Matern covariance function.
0081 model = stk_model('stk_materncov_aniso', DIM);
0082 
0083 % As a default choice, a constant (but unknown) mean is used, i.e.,  model.order = 0.
0084 % model.order = 1;  %%% UNCOMMENT TO USE A LINEAR TREND %%%
0085 % model.order = 2;  %%% UNCOMMENT TO USE A "FULL QUADRATIC" TREND %%%
0086 
0087 % Good practice: add a small "regularization noise" (nugget)
0088 MODEL_NOISE_STD = 1e-5;
0089 model.lognoisevariance = 2 * log(MODEL_NOISE_STD);
0090 
0091 
0092 %% EVALUATE THE FUNCTION ON A "MAXIMIN LHS" DESIGN
0093 
0094 xi = stk_sampling_maximinlhs(NI, DIM, BOX);
0095 zi = stk_feval(f, xi);
0096 
0097 % Simulate noisy evaluations (optional)
0098 TRUE_NOISE_STD = 0.0;
0099 if TRUE_NOISE_STD > 0
0100     zi = zi + randn(size(zi)) * TRUE_NOISE_STD;
0101 end
0102 
0103 % Add the design points to the first plot
0104 hold on; plot(xi(:,1), xi(:,2), DOT_STYLE{:});
0105 
0106 
0107 %% ESTIMATE THE PARAMETERS OF THE COVARIANCE FUNCTION
0108 
0109 % Initial guess for the parameters of the Matern covariance
0110 param0 = stk_param_init(model, xi, zi, BOX);
0111 
0112 % % Alternative: user-defined initial guess for the parameters of the Matern covariance
0113 % % (see "help stk_materncov_aniso" for more information)
0114 % SIGMA2 = var(zi);
0115 % NU     = 2;
0116 % RHO1   = (BOX(2,1) - BOX(1,1)) / 10;

0120 model.param = stk_param_estim(model, xi, zi, param0);
