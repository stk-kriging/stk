0036 %% DEFINE A 1D TEST FUNCTION
0037 
0038 f = @(x)( -(0.8*x + sin(5*x+1) + 0.1*sin(10*x)) );  % Define a 1D test function
0039 DIM = 1;                                            % Dimension of the factor space
0040 BOX = [-1.0; 1.0];                                  % Factor space
0041 
0042 NOISY = false;         % Choose either a noiseless or a noisy demo.
0043 if ~NOISY              % NOISELESS DEMO ...
0044     NI = 6;            %      NI = nb of evaluations that will be used
0045 else                   % OR NOISY DEMO ?
0046     NI = 20;           %      NI = nb of evaluations that will be used
0047     NOISESTD = 0.1;    %      NOISESTD = standard deviation of the observation noise
0048 end                    %
0049 
0050 
0051 %% "MAXIMIN LHS" SAMPLING PLAN & CORRESPONDING EVALUATIONS
0052 %
0053 % The objective is to construct an approximation of f with a budget of NI
0054 % evaluations performed on a "maximin LHS" design.
0055 %
0056 
0057 NITER = 5; % number of random designs generated in stk_sampling_maximinlhs()
0058 
0059 xi = stk_sampling_maximinlhs(NI, DIM, BOX, NITER);  % evaluation points
0060 %xi = stk_sampling_randunif(NI, DIM, BOX);
0061 zi = stk_feval(f, xi);                              % evaluation results
0062 
0063 if NOISY,
0064     zi = zi + NOISESTD * randn(NI, 1);
0065 end
0066 
0067 
0068 %% SPECIFICATION OF THE MODEL
0069 %
0070 % We choose a Matern covariance, the parameters of which will be estimated from the data.
0071 %
0072 
0073 % The following line defines a model with a constant but unknown mean (ordinary
0074 % kriging) and a Matern covariance function. (Some default parameters are also
0075 % set, but they will be replaced below by estimated parameters.)
0076 model = stk_model('stk_materncov_iso');
0077 
0078 
0079 %% ESTIMATION OF THE PARAMETERS OF THE COVARIANCE FUNCTION
0080 %
0081 % Here, the parameters of the Matern covariance function are estimated by the
0082 % REML (REstricted Maximum Likelihood) method using stk_param_estim().
0083 %
0084 
0085 % Initial guess for the parameters of the Matern covariance
0086 param0 = stk_param_init(model, xi, zi, BOX, NOISY);
0087 
0088 % % Alternative: user-defined initial guess for the parameters of the Matern covariance
0089 % % (see "help stk_materncov_iso" for more information)
0090 % SIGMA2 = 1.0;  % variance parameter
0091 % NU     = 4.0;  % regularity parameter
0092 % RHO1   = 0.4;  % scale (range) parameter
0093 % param0 = log([SIGMA2; NU; 1/RHO1]);
0094 
0095 if ~NOISY, % noiseless case
0096     model.lognoisevariance = 2 * log(1e-4); % small "regularization" noise (fixed)
0097     model.param = stk_param_estim(model, xi, zi, param0);
0098 else
0099     model.lognoisevariance = 2 * log(NOISESTD); % NOISESTD is assumed to be known
0100     model.param = stk_param_estim(model, xi, zi, param0);
0101 end
0102 
0103 model %#ok<NOPTS>
0104 
0105 
0106 %% CARRY OUT KRIGING PREDICTION AND DISPLAY RESULTS
0107 
0108 NT = 400;                                     % Number of points in the grid
0109 xt = stk_sampling_regulargrid(NT, DIM, BOX);  % Generate a regular grid of size NT
0110 zt = stk_feval(f, xt);                        % True value of the function on the grid
0111 
0112 % Compute the kriging predictor (and the kriging variance) on the grid
0113 zp = stk_predict(model, xi, zi, xt);
0114 
0115 % Visualisation
0116 stk_plot1d(xi, zi, xt, zt, zp);
0117 xlabel('x'); ylabel('z');
