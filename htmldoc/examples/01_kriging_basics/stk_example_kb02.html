0036 %% DEFINE A 1D TEST FUNCTION
0037 
0038 f = @(x)( -(0.8*x + sin(5*x+1) + 0.1*sin(10*x)) );  % Define a 1D test function
0039 DIM = 1;                                            % Dimension of the factor space
0040 BOX = [-1.0; 1.0];                                  % Factor space
0041 
0042 NOISY = false;         % Choose either a noiseless or a noisy demo.
0043 if ~NOISY              % NOISELESS DEMO ...
0044     NI = 6;            %      NI = nb of evaluations that will be used
0045 else                   % OR NOISY DEMO ?
0046     NI = 20;           %      NI = nb of evaluations that will be used
0047     NOISESTD = 0.1;    %      NOISESTD = standard deviation of the observation noise
0048 end                    %
0049 
0050 
0051 %% "MAXIMIN LHS" SAMPLING PLAN & CORRESPONDING EVALUATIONS
0052 %
0053 % The objective is to construct an approximation of f with a budget of NI
0054 % evaluations performed on a "maximin LHS" design.
0055 %
0056 
0057 NITER = 5; % number of random designs generated in stk_sampling_maximinlhs()
0058 
0059 xi = stk_sampling_maximinlhs(NI, DIM, BOX, NITER);  % evaluation points
0060 %xi = stk_sampling_randunif(NI, DIM, BOX);
0061 zi = stk_feval(f, xi);                              % evaluation results
0062 
0063 if NOISY,
0064     zi.a = zi.a + NOISESTD * randn(NI, 1);
0065     % (don't forget that the data is in the ".a" field!)
0066 end
0067 
0068 
0069 %% SPECIFICATION OF THE MODEL
0070 %
0071 % We choose a Matern covariance, the parameters of which will be estimated from the data.
0072 %
0073 
0074 % The following line defines a model with a constant but unknown mean (ordinary
0075 % kriging) and a Matern covariance function. (Some default parameters are also
0076 % set, but they will be replaced below by estimated parameters.)
0077 model = stk_model('stk_materncov_iso');
0078 
0079 
0080 %% ESTIMATION OF THE PARAMETERS OF THE COVARIANCE FUNCTION
0081 %
0082 % Here, the parameters of the Matern covariance function are estimated by the
0083 % REML (REstricted Maximum Likelihood) method using stk_param_estim().
0084 %
0085 
0086 % Initial guess for the parameters of the Matern covariance
0087 param0 = stk_param_init(model, xi, zi, BOX, NOISY);
0088 
0089 % % Alternative: user-defined initial guess for the parameters of the Matern covariance
0090 % % (see "help stk_materncov_iso" for more information)
0091 % SIGMA2 = 1.0;  % variance parameter
0092 % NU     = 4.0;  % regularity parameter
0093 % RHO1   = 0.4;  % scale (range) parameter
0094 % param0 = log([SIGMA2; NU; 1/RHO1]);
0095 
0096 if ~NOISY, % noiseless case
0097     model.lognoisevariance = 2 * log(1e-4); % small "regularization" noise (fixed)
0098     model.param = stk_param_estim(model, xi, zi, param0);
0099 else
0100     model.lognoisevariance = 2 * log(NOISESTD); % NOISESTD is assumed to be known
0101     model.param = stk_param_estim(model, xi, zi, param0);
0102 end
0103 
0104 model %#ok<NOPTS>
0105 
0106 
0107 %% CARRY OUT KRIGING PREDICTION AND DISPLAY RESULTS
0108 
0109 NT = 400;                                     % Number of points in the grid
0110 xt = stk_sampling_regulargrid(NT, DIM, BOX);  % Generate a regular grid of size NT
0111 zt = stk_feval(f, xt);                        % True value of the function on the grid
0112 
0113 % Compute the kriging predictor (and the kriging variance) on the grid
0114 zp = stk_predict(model, xi, zi, xt);
0115 
0116 % Visualisation
0117 stk_plot1d(xi, zi, xt, zt, zp);
