0032 %% Define the factor space & a Matern GP model on it
0033 
0034 % Factor space
0035 DIM = 1;
0036 BOX = [-1.0; 1.0];
0037 
0038 % Grid
0039 N = 400;
0040 x = stk_sampling_regulargrid (N, DIM, BOX);
0041 
0042 % Model
0043 model  = stk_model ('stk_materncov_iso');
0044 SIGMA2 = 1.0;  % variance parameter
0045 NU     = NaN;  % regularity parameter
0046 RHO1   = 0.4;  % scale (range) parameter
0047 model.param = log ([SIGMA2; NU; 1/RHO1]);
0048 
0049 % Several values for nu
0050 nu_list = [0.5 1.5 2.5 4.0];
0051 
0052 
0053 %% Generate (unconditional) sample paths
0054 
0055 NB_PATHS = 10;
0056 
0057 for k = 1:4,
0058     
0059     model.param(2) = log (nu_list(k));
0060     
0061     zsim = stk_generate_samplepaths (model, x, NB_PATHS);
0062     
0063     % Display the result
0064     stk_subplot (2, 2, k);  plot (x, zsim, 'LineWidth', 2);  legend off;
0065     stk_title (sprintf ('Matern, nu = %.1f', nu_list(k)));
0066     stk_labels ('input variable x', 'response z', 'FontWeight', 'bold');
0067     
0068 end
