0037 %% Define a 1d test function (the same as in example01.m)
0038 
0039 f = @(x)( -(0.8*x+sin(5*x+1)+0.1*sin(10*x)) );  % define a 1D test function
0040 DIM = 1;                                        % dimension of the factor space
0041 BOX = [-1.0; 1.0];                              % factor space
0042 
0043 NT = 400; % nb of points in the grid
0044 xt = stk_sampling_regulargrid(NT, DIM, BOX);
0045 zt = stk_feval(f, xt);
0046 
0047 
0048 %% Generate a random sampling plan
0049 %
0050 % The objective is to construct an approximation of f with a budget of NI
0051 % evaluations performed on a randomly generated (uniform) design.
0052 %
0053 % Change the value of NOISEVARIANCE to add a Gaussian evaluation noise on
0054 % the observations.
0055 %
0056 
0057 NOISEVARIANCE = 0.05;
0058 
0059 NI = 30;                                    % nb of evaluations that will be used
0060 xi = stk_sampling_randunif(NI, DIM, BOX);   % evaluation points
0061 zi = stk_feval(f, xi);                      % evaluation results
0062 
0063 zi = zi + sqrt(NOISEVARIANCE) * randn(NI,1);
0064 
0065 
0066 %% Specification of the model
0067 %
0068 % We choose a Matern covariance, the parameters of which will be estimated from the data.
0069 %
0070 % The values of the parameters that are provided here, including the noise variance, are
0071 % only used as an initial point for the optimization algorithm used in stk_param_estim().
0072 %
0073 
0074 % The following line defines a model with a constant but unknown mean (ordinary
0075 % kriging) and a Matern covariance function. (Some default parameters are also
0076 % set, but they will be replaced below by estimated parameters.)
0077 model = stk_model('stk_materncov_iso');
0078 
0079 % Noise variance
0080 model.lognoisevariance = log(100 * eps);
0081 % (this is not the true value of the noise variance !)
0082 
0083 
0084 %% Estimate the parameters of the covariance function
0085 %
0086 % Here, the parameters of the Matern covariance function are estimated by the
0087 % REML (REstricted Maximum Likelihood) method using stk_param_estim().
0088 %
0089 
0090 % Initial guess for the parameters for the Matern covariance
0091 % (see "help stk_materncov_iso" for more information)
0092 SIGMA2 = 1.0;  % variance parameter
0093 NU     = 4.0;  % regularity parameter
0094 RHO1   = 0.4;  % scale (range) parameter
0095 param0 = log([SIGMA2; NU; 1/RHO1]);
0096 
0097 % Initial guess for the (log of the) noise variance
0098 lnv0 = 2 * log(std(zi) / 100);
0099 
0100 [param, paramlnv] = stk_param_estim(model, xi, zi, param0, lnv0);
0101 
0102 model.param = param;
0103 model.lognoisevariance = paramlnv;
0104 
0105 
0106 %% Carry out kriging prediction
0107 
0108 zp = stk_predict(model, xi, zi, xt);
0109 
0110 
0111 %% Display results
0112 
0113 stk_plot1d(xi,zi,xt,zt,zp)
0114 xlabel('x'); ylabel('z');
