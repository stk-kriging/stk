0035 %% Define 1d test function (the same as in example01.m)
0036 
0037 f = @(x)( -(0.8*x+sin(5*x+1)+0.1*sin(10*x)) );  % define a 1D test function
0038 DIM = 1;                                        % dimension of the factor space
0039 BOX = [-1.0; 1.0];                              % factor space
0040 
0041 NT = 400; % nb of points in the grid
0042 xt = stk_sampling_regulargrid(NT, DIM, BOX);
0043 zt = stk_feval(f, xt);
0044 
0045 NI = 6;                                      % nb of evaluations that will be used
0046 xi = stk_sampling_randunif(NI, DIM, BOX);    % evaluation points
0047 zi = stk_feval(f, xi);                       % evaluation results
0048 
0049 
0050 %% Several Matern models
0051 
0052 NB_MODELS = 6; model = cell(1, NB_MODELS);
0053 
0054 % Parameters used as initial values for stk_param_estim()
0055 SIGMA2 = 1.0;  % variance parameter
0056 NU     = 2.0;  % regularity parameter
0057 RHO1   = 0.4;  % scale (range) parameter
0058 
0059 %%% first, two Matern models with estimated regularity
0060 
0061 % kriging with constant mean function (ordinary kriging)
0062 model{1} = stk_model('stk_materncov_iso');
0063 model{1}.param = log([SIGMA2; NU; 1/RHO1]);
0064 model{1}.lognoisevariance = log(100 * eps);
0065 
0066 % kriging with affine mean function
0067 model{2} = model{1};
0068 model{2}.order = 1;
0069 
0070 %%% two other Matern models with regularity parameter fixed to 3/2
0071 
0072 % kriging with constant mean function ("ordinary kriging)
0073 model{3} = stk_model('stk_materncov52_iso');
0074 model{3}.param = log([SIGMA2; log(1/RHO1)]);
0075 model{3}.lognoisevariance = log(100 * eps);
0076 
0077 % kriging with affine mean function
0078 model{4} = model{3};
0079 model{4}.order = 1;
0080 
0081 %%% two other Matern models with regularity parameter fixed to 5/2
0082 
0083 % kriging with constant mean function ("ordinary kriging)
0084 model{5} = stk_model('stk_materncov32_iso');
0085 model{5}.param = log([SIGMA2; 1/RHO1]);
0086 model{5}.lognoisevariance = log(100 * eps);
0087 
0088 % kriging with affine mean function
0089 model{6} = model{5};
0090 model{6}.order = 1;
0091 
0092 
0093 %% Parameter estimation and prediction for each model
0094 
0095 zp = cell(1, NB_MODELS);
0096 nc = floor(sqrt(NB_MODELS));
0097 nr = ceil(NB_MODELS / nc);
0098 
0099 for j = 1:NB_MODELS,
0100     % Estimate the parameters of the covariance
0101     model{j}.param = stk_param_estim(model{j}, xi, zi, model{j}.param);
0102     % Carry out kriging prediction
0103     zp{j} = stk_predict(model{j}, xi, zi, xt);
0104     % Plot the result
0105     h_axis = subplot(nr, nc, j);
0106     stk_plot1d(xi, zi, xt, zt, zp{j}, h_axis);
0107     % Title
0108     switch model{j}.covariance_type,
0109         case 'stk_materncov32_iso',
0110             title(sprintf('Matern 3/2, order=%d', model{j}.order));
0111         case 'stk_materncov52_iso',
0112             title(sprintf('Matern 5/2, order=%d', model{j}.order));
0113         case 'stk_materncov_iso',
0114             title(sprintf('Matern, estimated nu=%.2f, order=%d', ...
0115                 exp(model{j}.param(2)), model{j}.order));
0116     end
