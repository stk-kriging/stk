0032 %% DEFINE AN ARTIFICIAL ONE-DIMENSIONAL DATASET
0033 
0034 DIM = 1;            % dimension of the factor space
0035 BOX = [0.0; 1.0];   % factor space
0036 
0037 xi = [0.25; 0.26; 0.50; 0.60];
0038 zi = [1.00; 1.10; 0.20; 0.35];
0039 
0040 
0041 %% SPECIFICATION OF THE MODEL & REML ESTIMATION
0042 
0043 model = stk_model('stk_materncov_iso');
0044 
0045 % small "regularization noise".
0046 model.lognoisevariance = 2 * log(1e-6);
0047 
0048 % Mode of the prior on the parameters of the Matern covariance
0049 % (also serves as an initial guess for the optimization)
0050 SIGMA2 = 1.0;  % variance parameter
0051 NU     = 2.0;  % regularity parameter
0052 RHO1   = 0.4;  % scale (range) parameter
0053 param0 = log([SIGMA2; NU; 1/RHO1]);
0054 
0055 % Estimate covariance parameters (without prior)
0056 model.param = stk_param_estim(model, xi, zi, param0);
0057 param_opt_reml = model.param;
0058 
0059 
0060 %% EXPERIMENT WITH SEVERAL VALUES FOR PRIOR VARIANCES
0061 
0062 std_list = [10 1 0.2 0.01];
0063 
0064 NT = 400; % nb of points in the grid
0065 xt = stk_sampling_regulargrid(NT, DIM, BOX);
0066 
0067 figure('InvertHardcopy', 'off', 'Color', [1 1 1]);
0068 
0069 param_opt = zeros(length(param0), length(std_list));
0070 for k = 1:length(std_list),
0071     
0072     % Prior on the parameters of the Matern covariance
0073     model.prior.mean = param0;
0074     model.prior.invcov = eye(length(param0)) ./ (std_list(k)^2);
0075     
0076     % Estimate covariance parameters (with a prior)
0077     model.param = stk_param_estim(model, xi, zi, param0);
0078     param_opt(:, k) = model.param;
0079     
0080     % Carry out kriging prediction
0081     zp = stk_predict(model, xi, zi, xt);
0082     
0083     % Plot predicted values and pointwise confidences intervals
0084     haxis = subplot(2, 2, k); stk_plot1d(xi, zi, xt, [], zp, haxis);
0085     xlabel('input x'); ylabel('predicted output z');
0086     h = title(sprintf('prior std = %.2f', std_list(k)));
0087     set(h, 'FontWeight', 'bold');
0088 end
0089 
0090 
0091 %% FIGURE: ESTIMATED PARAMETER VERSUS PRIOR STD
0092 
0093 figure;
0094 param_name = {'SIGMA2', 'NU', '1/RHO'};
0095 for j = 1:3,
0096     subplot(2, 2, j);
0097     % estimated parameter versus prior std
0098     h = semilogx(std_list, exp(param_opt(j, :)), 'ko-');
0099     set(h, 'LineWidth', 2, 'MarkerFaceColor', 'y');
0100     xlabel('prior std'); ylabel(param_name{j});
0101     % add an horizontal line showing the value of REML estimation
0102     hold on; semilogx(xlim, exp(param_opt_reml(j))*[1 1], 'r--');
0103     % add a second horizontal line showing the mode of the prior
0104     hold on; semilogx(xlim, exp(param0(j))*[1 1], 'b--');
0105     % adjust ylim
0106     yy = exp([param_opt(j, :) param_opt_reml(j) param0(j)]);
0107     ylim_min = min(yy); ylim_max = max(yy); delta = ylim_max - ylim_min;
0108     ylim([ylim_min - 0.05*delta ylim_max + 0.05*delta]);
0109 end
