stk_materncov52_aniso

PURPOSE

STK_MATERNCOV52_ANISO computes the anisotropic Matern covariance with nu=5/2

SYNOPSIS

function k = stk_materncov52_aniso(param, x, y, diff, pairwise)

DESCRIPTION 

 STK_MATERNCOV52_ANISO computes the anisotropic Matern covariance with nu=5/2

 CALL: k = stk_materncov52_aniso(param, x, y, diff)
   param  = vector of parameters of size 1+d
   x      = structure whose field 'a' contains the observed points.
            x is a matrix of size n x d, where n is the number of
            points and d is the dimension of the factor space
   y      = same as x
   diff   = differentiation parameter

 STK_MATERNCOV52_ANISO computes a Matern covariance between two random vectors
 specified by the locations of the observations. This anisotropic
 covariance function has 2+d parameters, where d is the dimension of the
 factor space. They are defined as follows:
    param(1)   = log(sigma^2) is the logarithm of the variance
    param(1+i) = -log(rho(i)) is  the logarithm of  the inverse  of  the
                 i^th range parameter

 If diff ~= -1, the function returns the derivative of the covariance wrt
 param(diff)

0049 function k = stk_materncov52_aniso(param, x, y, diff, pairwise)
0050 if nargin > 5,
0051    stk_error ('Too many input arguments.', 'TooManyInputArgs');
0052 end
0053 
0054 persistent x0 y0 xs ys param0  pairwise0 D Kx_cache compute_Kx_cache
0055 
0056 % process input arguments
0057 x = double(x);
0058 y = double(y);
0059 if nargin < 4, diff = -1; end
0060 if nargin < 5, pairwise = false; end
0061 
0062 % check consistency for the number of factors
0063 dim = size(x, 2);
0064 if (size(y, 2) ~= dim),
0065     stk_error('xi and yi have incompatible sizes.', 'InvalidArgument');
0066 end
0067 nb_params = dim + 1;
0068 if (numel(param) ~= nb_params)
0069     stk_error('xi and param have incompatible sizes.', 'InvalidArgument');
0070 end
0071 
0072 % extract parameters from the "param" vector
0073 Sigma2 = exp(param(1));
0074 invRho = exp(param(2:end));
0075 
0076 % check parameter values
0077 if ~(Sigma2>0) || ~all(invRho>0),
0078     error('Incorrect parameter value.');
0079 end
0080 
0081 invRho = diag(invRho);
0082 
0083 % check if all input arguments are the same as before
0084 % (or if this is the first call to the function)
0085 if isempty(x0) || isempty(y0) || isempty(param0) || ...
0086         ~isequal({x, y, param}, {x0, y0, param0}) || ~isequal(pairwise, pairwise0)
0087     % compute the distance matrix
0088     xs = x * invRho; ys = y * invRho;
0089     D = stk_dist(xs, ys, pairwise);
0090     % save arguments for the next call
0091     x0 = x; y0 = y; param0 = param; pairwise0 = pairwise;
0092     % recomputation of Kx_cache is required
0093     compute_Kx_cache = true;
0094 end
0095 
0096 if diff == -1,
0097     %%% compute the value (not a derivative)
0098     k = Sigma2 * stk_sf_matern52(D, -1);
0099 elseif diff == 1,
0100     %%% diff wrt param(1) = log(Sigma2)
0101     k = Sigma2 * stk_sf_matern52(D, -1);
0102 elseif (diff >= 2) && (diff <= nb_params),
0103     %%% diff wrt param(diff) = - log(invRho(diff-1))
0104     ind = diff - 1;
0105     if compute_Kx_cache || isempty(Kx_cache)
0106         Kx_cache  = 1./(D+eps) .* (Sigma2 * stk_sf_matern52(D, 1));
0107         compute_Kx_cache = false;
0108     end
0109     nx = size(x, 1); ny = size(y, 1);
0110     if pairwise,
0111         k = (xs(:, ind) - ys(:, ind)).^2 .* Kx_cache;
0112     else
0113         k = (repmat(xs(:, ind), 1, ny) - repmat(ys(:, ind)', nx, 1)).^2 .* Kx_cache;
0114     end
0115 else
0116     stk_error('Incorrect value for the ''diff'' parameter.', 'InvalidArgument');
0117 end
0118 
0119 end % function
