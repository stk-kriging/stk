stk_materncov_aniso

PURPOSE

STK_MATERNCOV_ANISO computes the anisotropic Matern covariance

SYNOPSIS

function k = stk_materncov_aniso(param, x, y, diff)

DESCRIPTION 

 STK_MATERNCOV_ANISO computes the anisotropic Matern covariance

 CALL: k = stk_materncov_aniso(param, x, y, diff)
   param  = vector of parameters of size 2+d
   x      = structure whose field 'a' contains the observed points.
            x.a  is a matrix of size n x d, where n is the number of
            points and d is the dimension of the factor space
   y      = same as x
   diff   = differentiation parameter

 STK_MATERNCOV_ANISO computes a Matern covariance between two random vectors
 specified by the locations of the observations. This anisotropic
 covariance function has 2+d parameters, where d is the dimension of the
 factor space. They are defined as follows:
    param(1)   = log(sigma^2) is the logarithm of the variance
    param(2)   = log(nu) is the logarithm of the regularity parameter
    param(2+i) = -log(rho(i)) is the logarithm of the inverse of the
                 range parameter for the ith dimension

 If diff ~= -1, the function returns the derivative of the covariance wrt
 param(diff)

0054 stk_narginchk(3, 4);
0055 
0056 % default: compute the value (not a derivative)
0057 if (nargin<4), diff = -1; end

0059 % check size consistency
0060 dim = size(x.a, 2);
0061 if (size(y.a, 2) ~= dim),
0062     stk_error('xi.a and yi.a have incompatible sizes.', 'InvalidArgument');
0063 end
0064 nb_params = dim + 2;
0065 if (numel(param) ~= nb_params)
0066     stk_error('xi.a and param have incompatible sizes.', 'InvalidArgument');
0067 end
0068 
0069 % extract parameters from the "param" vector
0070 Sigma2 = exp(param(1));
0071 Nu     = exp(param(2));
0072 invRho = exp(param(3:end));
0073 
0074 % check parameter values
0075 if ~(Sigma2>0) || ~(Nu>0) || ~all(invRho>0),
0076     error('Incorrect parameter value.');
0077 end
0078 
0079 invRho = diag(invRho);
0080 
0081 % check if all input arguments are the same as before
0082 % (or if this is the first call to the function)
0083 if isempty(x0) || isempty(y0) || isempty(param0) || ...
0084         ~isequal({x.a,y.a,param},{x0.a,y0.a,param0})
0085     % compute the distance matrix
0086     xs = x.a * invRho; ys = y.a * invRho;
0087     D = stk_distance_matrix(xs, ys);
0088     % save arguments for the next call
0089     x0 = x; y0 = y; param0 = param;
0090     % recomputation of Kx_cache is required
0091     compute_Kx_cache = true;
0092 end
0093 
0094 if diff == -1,
0095     %%% compute the value (not a derivative)
0096     k = Sigma2 * stk_sf_matern(Nu, D, -1);
0097 elseif diff == 1,
0098     %%% diff wrt param(1) = log(Sigma2)
0099     k = Sigma2 * stk_sf_matern(Nu, D, -1);
0100 elseif diff == 2,
0101     %%% diff wrt param(2) = log(Nu)
0102     k = Nu * Sigma2 * stk_sf_matern(Nu, D, 1);
0103 elseif (diff >= 3) && (diff <= nb_params),
0104     %%% diff wrt param(diff) = - log(invRho(diff-2))
0105     ind = diff - 2;
0106     if compute_Kx_cache || isempty(Kx_cache)
0107         Kx_cache  = 1./(D+eps) .* (Sigma2 * stk_sf_matern(Nu, D, 2));
0108         compute_Kx_cache = false;
0109     end
0110     nx = size(x.a,1); ny = size(y.a,1);
0111     k = (repmat(xs(:,ind),1,ny) - repmat(ys(:,ind)',nx,1)).^2 .* Kx_cache;
0112 else
0113     stk_error('Incorrect value for the ''diff'' parameter.', 'InvalidArgument');
0114 end
