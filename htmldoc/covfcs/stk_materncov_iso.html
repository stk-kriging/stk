stk_materncov_iso

PURPOSE

STK_MATERNCOV_ISO computes the isotropic Matern covariance

SYNOPSIS

function k = stk_materncov_iso(param, x, y, diff, pairwise)

DESCRIPTION 

 STK_MATERNCOV_ISO computes the isotropic Matern covariance

 CALL: [k]=stk_materncov_iso(param, x, y, diff)
   param  = vector of parameters of size 3
   x      = structure whose field 'a' contains the observed points.
            x is a matrix of size n x d, where n is the number of
            points and d is the dimension of the
            factor space
   y      = same as x
   diff   = differentiation parameter

 STK_MATERNCOV_ISO computes a Matern covariance between two random vectors
 specified by the locations of the observations. The isotropic covariance
 function has three parameters
    param(1) = log(sigma^2) is the logarithm of the variance of random
               process
    param(2) = log(nu) is the logarithm of the regularity parameter of the
               Matern covariance
    param(3) = -log(rho) is the logarithm of the inverse of the range
               parameter
 If diff ~= -1, the function returns the derivative of the covariance wrt
 param(diff)

0051 function k = stk_materncov_iso(param, x, y, diff, pairwise)
0052 if nargin > 5,
0053    stk_error ('Too many input arguments.', 'TooManyInputArgs');
0054 end
0055 
0056 persistent x0 y0 param0 pairwise0 D
0057 
0058 % process input arguments
0059 x = double(x);
0060 y = double(y);
0061 if nargin < 4, diff = -1; end
0062 if nargin < 5, pairwise = false; end
0063 
0064 % extract parameters from the "param" vector
0065 Sigma2 = exp(param(1));
0066 Nu     = exp(param(2));
0067 invRho = exp(param(3));
0068 
0069 % check parameter values
0070 if ~(Sigma2>0) || ~(Nu>0) || ~(invRho>0),
0071     error('Incorrect parameter value.');
0072 end
0073 
0074 % check if all input arguments are the same as before
0075 % (or if this is the first call to the function)
0076 if isempty(x0) || isempty(y0) || isempty(param0) || ...
0077         ~isequal({x, y, param}, {x0, y0, param0}) || ~isequal(pairwise, pairwise0)
0078     % compute the distance matrix
0079     D  = invRho * stk_dist(x, y, pairwise);
0080     % save arguments for the nex call
0081     x0 = x; y0 = y; param0 = param; pairwise0 = pairwise;
0082 end
0083 
0084 if diff == -1,
0085     %%% compute the value (not a derivative)
0086     k = Sigma2 * stk_sf_matern(Nu, D, -1);
0087 elseif diff == 1,
0088     %%% diff wrt param(1) = log(Sigma2)
0089     k = Sigma2 * stk_sf_matern(Nu, D, -1);
0090 elseif diff == 2,
0091     %%% diff wrt param(2) = log(Nu)
0092     k = Nu * Sigma2 * stk_sf_matern(Nu, D, 1);
0093 elseif diff == 3,
0094     %%% diff wrt param(3) = - log(invRho)
0095     k = D .* (Sigma2 * stk_sf_matern(Nu, D, 2));
0096 else
0097     error('there must be something wrong here !');
0098 end
