0030 function K = stk_discretecov (param, x, y, diff, pairwise)
0031 
0032 %--- process input arguments ---------------------------------------------------
0033 
0034 x = double (x);
0035 y = double (y);
0036 
0037 if (nargin >= 4) && (diff ~= -1),
0038     stk_error ('diff should be equal to -1', 'InvalidArgument');
0039 end
0040 
0041 if nargin < 5,
0042     pairwise = false;
0043 end
0044 
0045 %--- compute covariance matrix -------------------------------------------------
0046 
0047 if ~pairwise,
0048     K = param.K(x, y);
0049 else
0050     idx = sub2ind (size (param.K), x, y);
0051     K = param.K(idx);
0052 end
0053     
0054 end % function
