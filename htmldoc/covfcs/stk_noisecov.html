0030 function K = stk_noisecov (ni, lognoisevariance, diff)
0031 
0032 if nargin > 3,
0033    stk_error ('Too many input arguments.', 'TooManyInputArgs');
0034 end
0035 
0036 if nargin < 3,
0037     diff = -1; % default: compute the value (not a derivative)
0038 end
0039 
0040 if isscalar (lognoisevariance), % homoscedastic
0041     
0042     % the result does not depend on diff
0043     K = exp(lognoisevariance) * eye(ni);
0044     
0045 else % heteroscedastic
0046     
0047     if ~ ((isequal (s, [1, ni])) || (isequal (s, [ni, 1])))
0048         error ('lognoisevariance must be a scalar or a vector of length ni.');
0049     end
0050 
0051     if diff ~= -1,
0052         error ('diff ~= -1 is not allowed in the heteroscedastic case');
0053     end
0054     
0055     K = diag (exp (lognoisevariance));
0056 end
