0030 function K = stk_noisecov(ni, lognoisevariance, diff)
0031 
0032 stk_narginchk(2, 3);
0033 
0034 s = size(lognoisevariance);
0035 n = max(s);
0036 if ~isequal(s, [1,n]) && ~isequal(s, [n,1])
0037     error('lognoisvariance must be a vector.');
0038 end
0039 
0040 if nargin == 2,
0041     diff = -1; % default: compute the value (not a derivative)
0042 end
0043 
0044 if n == 1
0045     % the result does not depend on diff
0046     K = exp(lognoisevariance) * eye(ni);
0047 else
0048     if diff ~= -1,
0049         error('not implemented');
0050     end
0051     K = diag(exp(lognoisevariance));
0052 end
0053 
0054 end % function
