0029 function y = stk_datastruct(x)
0030 stk_narginchk(0, 1);
0031 
0032 if (nargin == 0) || isempty(x)
0033     y = struct('a', []);
0034     return;
0035 end
0036     
0037 if isstruct(x) && isfield(x, 'a')
0038     y = x;
0039     return;
0040 end
0041 
0042 try
0043     y = struct('a', double(x));
0044 catch
0045     xname = inputname(1);
0046     if isempty(xname),
0047         errmsg = 'x must be a matrix or a structure with an ''a'' field.';
0048         stk_error(errmsg, 'IncorrectArgument');
0049     else
0050         stack = dbstack();
0051         errmsg = sprintf('%s must be a matrix or a structure with an ''a'' field.', xname);
0052         stk_error(errmsg, 'IncorrectArgument', stack(2:end));
0053     end
0054 end
