0029 function t = subsref(x, idx)
0030 
0031 switch idx(1).type
0032     
0033     case '()'
0034         
0035         if length(idx) ~= 1
0036             
0037             stk_error('Illegal indexing.', 'IllegalIndexing');
0038             
0039         else % ok, only one level of indexing
0040             
0041             [n, d] = size (x);
0042             L = length (idx(1).subs);
0043             
0044             if d == 1, % univariate dataframe
0045                 
0046                 if L == 1,
0047                     % linear indexing allowed for univariate dataframes
0048                     I = idx(1).subs{1};
0049                     J = 1;
0050                 else
0051                     % general case: matrix-style indexing expected
0052                     if L ~= 2,
0053                         stk_error(['Illegal indexing for a univariate ' ...
0054                             'stk_dataframe object.'], 'IllegalIndexing');
0055                     else
0056                         I = idx(1).subs{1};
0057                         J = idx(1).subs{2};
0058                     end
0059                 end
0060                 
0061             else % multivariate dataframe: matrix-style indexing expected
0062                 
0063                 if L ~= 2,
0064                     stk_error(['multivariate stk_dataframe objects only ' ...
0065                         'support matrix-style indexing.'], 'IllegalIndexing');
0066                 else
0067                     I = idx(1).subs{1};
0068                     J = idx(1).subs{2};
0069                 end
0070                 
0071             end
0072             
0073             c = get (x, 'colnames');
0074             if ~ isempty (c),
0075                 c = c(1, J);
0076             end
0077             
0078             r = get (x, 'rownames');
0079             if ~ isempty (r),
0080                 r = r(I, 1);
0081             end
0082             
0083             t = stk_dataframe (x.data(I, J), c, r);
0084             
0085         end
0086         
0087     case '{}'
0088         
0089         errmsg = 'Indexing with curly braces is not allowed.';
0090         stk_error(errmsg, 'IllegalIndexing');
0091         
0092     case '.'
0093         
0094         t = get(x, idx(1).subs);
0095         
0096         if length(idx) > 1,
0097             t = subsref(t, idx(2:end));
0098         end
0099         
0100 end
