0029 function z = horzcat(x, y, varargin)
0030 
0031 if nargin < 2,
0032     y = [];
0033 end
0034 
0035 if isa(x, 'stk_dataframe')
0036     
0037     % In this case, [x y] will be an stk_dataframe also.
0038     
0039     x_data = double (x);
0040     y_data = double (y);
0041         
0042     if isa(y, 'stk_dataframe')
0043         y_colnames = get (y, 'colnames');
0044         y_rownames = get (y, 'rownames');
0045     else
0046         y_colnames = {};
0047         y_rownames = {};
0048     end
0049     
0050     %--- ROW NAMES --------------------------------------------------------
0051 
0052     x_rownames = get (x, 'rownames');
0053     
0054     if isempty(x_rownames)
0055         
0056         rownames = y_rownames;
0057         
0058     elseif ~isempty(y_rownames) && ~all(strcmp(x_rownames, y_rownames))
0059         
0060         stk_error(['Cannot concatenate because of incompatible row ' ...
0061             'names.'], 'IncompatibleRowColNames');
0062         
0063     else % ok, we can use x's column names
0064         
0065         rownames = x_rownames;
0066         
0067     end
0068     
0069     %--- COLUMN NAMES -----------------------------------------------------
0070 
0071     x_colnames = get (x, 'colnames');
0072     
0073     bx = isempty(x_colnames);
0074     by = isempty(y_colnames);
0075     
0076     if bx && by % none of the argument has row names
0077         
0078         colnames = {};
0079         
0080     else % at least of one the arguments has column names
0081        
0082         if bx
0083             x_colnames = repmat({''}, 1, size(x_data, 2));
0084         end
0085         
0086         if by
0087             y_colnames = repmat({''}, 1, size(y_data, 2));
0088         end
0089         
0090         colnames = [x_colnames y_colnames];
0091         
0092     end
0093     
0094     z = stk_dataframe([x_data y_data], colnames, rownames);
0095     
0096 else % In this case, z will be a matrix.
0097     
0098     z = [double(x) double(y)];
0099 
0100 end
0101 
0102 if ~isempty(varargin),
0103     z = horzcat(z, varargin{:});
0104 end
0105
