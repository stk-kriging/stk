0029 function z = vertcat(x, y, varargin)
0030 
0031 if isa(x, 'stk_dataframe')
0032     
0033     % In this case, [x; y] will be an stk_dataframe also.
0034     
0035     y_data = double(y);
0036     data = [double(x); y_data];
0037     
0038     if isa(y, 'stk_dataframe')
0039         y_colnames = get (y, 'colnames');
0040         y_rownames = get (y, 'rownames');
0041     else
0042         y_colnames = {};
0043         y_rownames = {};
0044     end
0045     
0046     %--- COLUMN NAMES -----------------------------------------------------
0047     
0048     x_colnames = get (x, 'colnames');
0049     
0050     if isempty (x_colnames)
0051         
0052         colnames = y_colnames;
0053         
0054     elseif ~isempty(y_colnames) && ~all(strcmp(x_colnames, y_colnames))
0055         
0056         stk_error(['Cannot concatenate because of incompatible column ' ...
0057             'names.'], 'IncompatibleColNames');
0058         
0059     else % ok, we can use x's column names
0060         
0061         colnames = x_colnames;
0062         
0063     end
0064     
0065     %--- ROW NAMES --------------------------------------------------------
0066     
0067     x_rownames = get (x, 'rownames');
0068     
0069     bx = isempty(x_rownames);
0070     by = isempty(y_rownames);
0071     
0072     if bx && by, % none of the argument has row names
0073         
0074         rownames = {};
0075         
0076     else % at least of one the arguments has row names
0077         
0078         if bx
0079             x_rownames = repmat({''}, size(x.data, 1), 1);
0080         end
0081         
0082         if by
0083             y_rownames = repmat({''}, size(y_data, 1), 1);
0084         end
0085         
0086         rownames = [x_rownames; y_rownames];
0087         
0088     end
0089     
0090     z = stk_dataframe(data, colnames, rownames);
0091     
0092 else  % In this case, z will be a matrix.
0093     
0094     z = [double(x); double(y)];
0095     
0096 end
0097 
0098 if ~isempty(varargin),
0099     z = vertcat(z, varargin{:});
0100 end
