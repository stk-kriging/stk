0029 function y = bsxfun(F, x1, x2)
0030 
0031 x1data = double (x1);
0032 x2data = double (x2);
0033 
0034 try
0035     ydata = bsxfun(F, x1data, x2data);
0036 catch
0037     err = lasterror ();
0038     if strcmp (err.identifier, 'MATLAB:bsxfun:unsupportedBuiltin')
0039         % This happens in some old versions of Matlab with realpow, for
0040         % instance. Let's try without singleton expansion...
0041         ydata = feval (F, x1data, x2data);
0042         % TODO: manual bsxfun !!!
0043     else
0044         rethrow (err);
0045     end
0046 end
0047 
0048 % choose if the output type
0049 if isa(x1, 'stk_dataframe'),
0050     y = stk_dataframe(ydata, x1.colnames);
0051 else
0052     y = ydata;
0053 end
