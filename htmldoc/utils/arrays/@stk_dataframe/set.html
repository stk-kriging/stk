0029 function x = set (x, propname, value)
0030 
0031 icol = get_column_number (x.colnames, propname);
0032 
0033 switch icol
0034     
0035     case -4 % 'info'
0036         
0037         x.info = value;
0038         
0039     case -3 % 'rownames'
0040         
0041         if isempty (value)
0042             x.rownames = {};
0043         else
0044             n1 = size (x.data, 1);
0045             n2 = length (value);
0046             if ~ iscell (value) && (isequal (size (value), [1 n2]) ...
0047                     || isequal (value, [n2 1])),
0048                 stk_error ('value should be a vector-shaped cell array.');
0049             else
0050                 value = value(:);
0051                 b = cellfun (@isempty, value);
0052                 value(b) = repmat ({''}, sum (b), 1);
0053                 if n2 <= n1
0054                     x.rownames = [value; repmat({''}, n1 - n2, 1)];
0055                 else
0056                     x.rownames = value;
0057                     x.data = [x.data; nan(n2 - n1, size(x.data, 2))];
0058                 end
0059             end
0060         end
0061         
0062     case -2 % 'colnames'
0063         
0064         if isempty (value)
0065             x.colnames = {};
0066         else
0067             d1 = size (x.data, 2);
0068             d2 = length (value);
0069             if ~ iscell (value) && (isequal (size (value), [1 d2]) ...
0070                     || isequal (value, [d2 1])),
0071                 stk_error ('value should be a vector-shaped cell array.');
0072             else
0073                 value = reshape (value, 1, d2);
0074                 b = cellfun (@isempty, value);
0075                 value(b) = repmat ({''}, 1, sum (b));
0076                 if d2 <= d1
0077                     x.colnames = [value repmat({''}, 1, d1 - d2)];
0078                 else
0079                     x.colnames = value;
0080                     x.data = [x.data nan(size(x.data, 1), d2 - d1)];
0081                 end
0082             end
0083         end
0084         
0085     case - 1 % set entire array
0086         
0087         [n1, d1] = size (x.data);
0088         [n2, d2] = size (value);
0089         x.data = value;
0090         
0091         if (n1 ~= n2) && ~ isempty (x.rownames)
0092             if n2 > n1,
0093                 % silently add rows without a name
0094                 x.rownames = [x.rownames; repmat({''}, n2 - n1, 1)];
0095             else
0096                 % delete superfluous row names and emit a warning
0097                 x.rownames = x.rownames(1:n2);
0098                 warning ('Some row names have been deleted.');
0099             end
0100         end
0101         
0102         if (d1 ~= d2) && ~ isempty (x.colnames)
0103             if d2 > d1,
0104                 % silently add columns without a name
0105                 x.colnames = [x.colnames; repmat({''}, 1, d2 - d1)];
0106             else
0107                 % delete superfluous column names and emit a warning
0108                 x.colnames = x.colnames(1:d2);
0109                 warning ('Some column names have been deleted.');
0110             end
0111         end
0112         
0113     otherwise
0114         if isequal (size(value), [size(x.data, 1) 1])
0115             x.data(:, icol) = value;
0116         else
0117             error ('Incorrect size');
0118         end
0119         
0120 end
0121 
0122 end % function set
