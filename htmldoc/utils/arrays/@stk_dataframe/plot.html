0029 function plot (x, varargin)
0030 
0031 if ~ strcmp (x, 'stk_dataframe'),
0032     if isa (x, 'stk_dataframe')
0033         x = stk_dataframe (double (x), get (x, 'colnames'));
0034     else
0035         x = stk_dataframe (double (x));
0036     end
0037 end
0038 
0039 if (nargin > 1) && (~ ischar (varargin{1}))
0040     if size(x, 2) > 1,
0041         stk_error('Incorrect size for argument x.', 'IncorrectSize');
0042     else
0043         z = varargin{1};
0044         n = size (x, 1);
0045         if size (z, 1) == n,
0046             x = horzcat (x, z);
0047         elseif isequal(size (z), [1 n]),
0048             % special case: we tolerate that in silence...
0049             x = horzcat (x, z');
0050         end
0051         opts_pos = 2;
0052     end
0053 else
0054     if size (x, 2) > 2,
0055         warning ('Plotting the first two columns only.'); %#ok<WNTAG>
0056         % TODO: implement scatter plot matrices
0057     end
0058     opts_pos = 1;
0059 end
0060 
0061 if nargin > opts_pos,
0062     opts = varargin(opts_pos:end);
0063 else
0064     opts = {};
0065 end
0066 
0067 xx = double (x);
0068 yy = xx(:, 2:end);
0069 xx = xx(:, 1);
0070 
0071 nb_outputs = size (yy, 2);
0072 
0073 colnames = get (x, 'colnames');
0074 
0075 if ~ isempty (colnames)
0076     xlab = colnames{1};
0077     ylab = colnames(2:end);
0078 else
0079     xlab = '';
0080     ylab = {};
0081 end
0082 
0083 plot (xx, yy, opts{:});
0084 
0085 if ~ isempty (xlab),
0086     stk_xlabel (xlab);
0087 end
0088 
0089 if ~ isempty (ylab)
0090     if nb_outputs == 1,
0091         stk_ylabel (ylab{1});
0092     else
0093         legend (ylab{:});
0094     end
0095 end
0096 
0097 end % function plot
