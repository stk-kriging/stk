0029 function s = stk_sprintf_levels (x)
0030 
0031 levels = x.levels;
0032 colnames = get (x.stk_dataframe, 'colnames');
0033 
0034 d = length (levels);
0035 s = sprintf ('1 x %d cell array', d);
0036 
0037 for i = 1:d   
0038     
0039     if isempty (colnames)
0040         line{i} = sprintf('levels for column #%d: ', i);
0041     else
0042         line{i} = sprintf('levels for variable %s: ', colnames{i});
0043     end
0044         
0045     L = levels{i};
0046     if isempty (L)
0047         line{i} = [line{i} '[]'];
0048     else
0049         for j = 1:(length(L) - 1)
0050             line{i} = [line{i} num2str(L(j)) ', '];
0051         end
0052         line{i} = [line{i} sprintf('%s', num2str(L(end)))];
0053     end
0054     
0055 end
0056 
0057 s = strvcat (s, line{:});
0058 
0059 end % function stk_sprintf_levels
