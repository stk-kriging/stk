0030 function s = stk_sprintf_data (x, max_width, colnames, rownames)
0031 
0032 if ~ isnumeric (x)
0033     
0034     errmsg = sprintf ('Incorrect argument type: %s', class (x));
0035     stk_error (errmsg, 'IncorrectType');
0036     
0037 else
0038     
0039     x = double (x);
0040     [n, d] = size (x);
0041     
0042     if (n == 0) || (d == 0)
0043         
0044         s = '[] (empty)';
0045         
0046     else
0047                 
0048         if (nargin < 2) || isempty (max_width)
0049             try
0050                 switch get (0, 'Format')
0051                     case 'short'
0052                         max_width = 6;
0053                     case 'long'
0054                         max_width = 16;
0055                     otherwise
0056                         % FIXME: handle other formatting modes...
0057                         max_width = 10;
0058                 end
0059             catch
0060                 % Property 'Format' doesn't exist in Octave 3.2.x
0061                 max_width = 6;
0062             end
0063         end
0064 
0065         % column names
0066         if (nargin < 3) || isempty (colnames)
0067             colnames = repmat ({''}, 1, d);
0068         end
0069         
0070         % row names
0071         if (nargin < 4) || isempty (rownames)
0072             rownames = repmat ({''}, n, 1);
0073         end
0074         b = cellfun (@isempty, rownames);
0075         rownames(b) = repmat({'*'}, sum (b), 1);
0076         rownames = fliplr (char (cellfun ...
0077             (@fliplr, rownames, 'UniformOutput', false)));
0078         
0079         nb_spaces_colsep = 2;
0080         
0081         nb_rows = n + 1;  % + 1 for the header
0082         
0083         s = repmat ('', nb_rows, 1); %#ok<*AGROW>
0084                 
0085         s = [s [repmat(' ', 1, size(rownames, 2)); rownames]];
0086         
0087         % column separator between row names and the first data column
0088         s = [s repmat(' :', nb_rows,  1) ...
0089             repmat(' ', nb_rows,  nb_spaces_colsep)];
0090         
0091         for j = 1:d,
0092             
0093             xx = stk_sprintf_colvect (x(:, j), max_width);
0094             Lxx = size (xx, 2);
0095             
0096             vn = colnames{j};
0097             if isempty (vn),
0098                 vn = repmat('-', 1, Lxx);
0099             end
0100             
0101             L = max (length (vn), Lxx);
0102             s = [s [sprintf('% *s', L, vn);   % variable name
0103                 repmat(' ', n, L - Lxx) xx]]; % formatted data
0104             
0105             if j < d,
0106                 % column separator
0107                 s = [s repmat(' ', nb_rows,  nb_spaces_colsep)];
0108             end
0109             
0110         end % for
0111         
0112     end % if
0113     
0114 end % function stk_sprintf_data
