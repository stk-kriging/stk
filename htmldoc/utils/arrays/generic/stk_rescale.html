0029 function [y, a, b] = stk_rescale(x, box1, box2)
0030 
0031 if nargin > 3,
0032    stk_error ('Too many input arguments.', 'TooManyInputArgs');
0033 end
0034 
0035 % read argument x
0036 x = double(x);
0037 [n, d] = size(x);
0038 
0039 % read box1
0040 if ~isempty(box1),
0041     stk_assert_box(box1, d);
0042 end
0043 
0044 % read box2
0045 if ~isempty(box2),
0046     stk_assert_box(box2, d);
0047 end
0048 
0049 % scale to [0; 1] (xx --> zz)
0050 if ~isempty(box1),
0051     xmin = box1(1, :);
0052     xmax = box1(2, :);
0053     b1 = 1 ./ (xmax - xmin);
0054     a1 = - xmin .* b1;
0055 else
0056     b1 = ones(1, d);
0057     a1 = zeros(1, d);
0058 end
0059 
0060 % scale to box2 (zz --> yy)
0061 if ~isempty(box2),
0062     zmin = box2(1, :);
0063     zmax = box2(2, :);
0064     b2 = zmax - zmin;
0065     a2 = zmin;
0066 else
0067     b2 = ones(1, d);
0068     a2 = zeros(1, d);    
0069 end
0070 
0071 b = b2 .* b1;
0072 a = a2 + a1 .* b2;
0073 y = repmat(a, n, 1) + x * diag(b);
0074 
0075 end % function stk_rescale
