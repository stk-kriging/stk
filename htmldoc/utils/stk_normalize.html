0029 function y = stk_normalize(x, box)
0030 
0031 stk_narginchk(1, 2);
0032 
0033 y = stk_datastruct(x);
0034 n = size(y.a, 1);
0035 
0036 if nargin < 2,
0037     xmin = min(y.a, [], 1);
0038     xmax = max(y.a, [], 1);
0039 else
0040     xmin = box(1, :);
0041     xmax = box(2, :);
0042 end
0043 
0044 y.a = (y.a - repmat(xmin, n, 1)) ./ repmat(xmax - xmin, n, 1);
0045 
0046 end % function stk_normalize
