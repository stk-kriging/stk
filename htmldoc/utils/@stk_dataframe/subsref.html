0029 function t = subsref(x, idx)
0030 
0031 switch idx(1).type
0032     
0033     case '()'
0034         
0035         if length(idx) ~= 1
0036             
0037             stk_error('Illegal indexing.', 'IllegalIndexing');
0038             
0039         else % ok, only one level of indexing
0040             
0041             d = size(x, 2);
0042             L = length(idx(1).subs);
0043             
0044             if (d == 1) && ~((L == 1) || (L == 2))
0045                 
0046                 stk_error(['Illegal indexing for a univariate stk_dataframe' ...
0047                     'object.'], 'IllegalIndexing');
0048                 
0049             elseif (d > 1) && (L ~= 2)
0050                 
0051                 stk_error(['multivariate stk_dataframe objects only support ' ...
0052                     'matrix-style indexing.'], 'IllegalIndexing');
0053                 
0054             else % ok, legal indexing
0055                 
0056                 t = subsref(x.data, idx);
0057                 
0058             end
0059             
0060         end
0061         
0062     case '{}'
0063         
0064         errmsg = 'Indexing with curly braces is not allowed.';
0065         stk_error(errmsg, 'IllegalIndexing');
0066         
0067     case '.'
0068         
0069         t = get(x, idx(1).subs);
0070         
0071         if length(idx) > 1,
0072             t = subsref(t, idx(2:end));
0073         end
0074         
0075 end
