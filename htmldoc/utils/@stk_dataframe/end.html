0029 function idx = end(x, k, nb_indices)
0030 
0031 if nb_indices == 2,    
0032     % using two indices (matrix-style indexing)
0033     idx = size(x.data, k);    
0034 else    
0035     if nb_indices ~= 1,
0036         errmsg = 'stk_dataframe objects only support linear or matrix-style indexing.';
0037         stk_error(errmsg, 'IllegalIndexing');
0038     else
0039         % using linear indexing
0040         idx = numel(x.data);
0041     end
0042 end
