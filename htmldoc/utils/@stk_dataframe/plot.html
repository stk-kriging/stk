0029 function plot(x, z, varargin)
0030 
0031 xx = double(x);
0032 zz = double(z);
0033 
0034 if size(xx, 2) > 1,
0035     
0036     stk_error('Incorrect size for argument x.', 'IncorrectSize');
0037     
0038 else % ok, the first argument has one (and only one) column
0039     
0040     plot(xx, zz, varargin{:});
0041     
0042     if isa(x, 'stk_dataframe') && ~isempty(x.vnames)
0043         xlabel(x.vnames{1}, 'FontWeight', 'bold');
0044     end
0045     
0046     if isa(z, 'stk_dataframe') && ~isempty(z.vnames)
0047         if size(zz, 2) == 1,
0048             ylabel(z.vnames{1}, 'FontWeight', 'bold');
0049         else
0050             legend(z.vnames);
0051         end
0052     end
0053     
0054 end % if
0055 
0056 end % function plot
