0029 function y = stk_rescale(x, box1, box2)
0030 stk_narginchk(3, 3);
0031 
0032 % read argument x
0033 if isstruct(x),
0034     xx = x.a;
0035 else
0036     xx = x;
0037 end
0038 [n, d] = size(xx);
0039 
0040 % read box1
0041 if ~isempty(box1),
0042     stk_assert_box(box1, d);
0043 end
0044 
0045 % read box2
0046 if ~isempty(box2),
0047     stk_assert_box(box2, d);
0048 end
0049 
0050 % scale to [0; 1] (xx --> zz)
0051 if ~isempty(box1),
0052     xmin = box1(1, :);
0053     xmax = box1(2, :);
0054     delta = xmax - xmin;
0055     zz = (xx - repmat(xmin, n, 1)) .* repmat(1./delta, n, 1);
0056 else
0057     zz = xx;
0058 end
0059 
0060 % scale to box2 (zz --> yy)
0061 if ~isempty(box2),
0062     zmin = box2(1, :);
0063     zmax = box2(2, :);
0064     delta = zmax - zmin;
0065     yy = repmat(zmin, n, 1) + zz .* repmat(delta, n, 1);
0066 else
0067     yy = zz;
0068 end
0069 
0070 % output
0071 if isstruct(x),
0072     y = x;
0073     y.a = yy;
0074 else
0075     y = yy;
0076 end
