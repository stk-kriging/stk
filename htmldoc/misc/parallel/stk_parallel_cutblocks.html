0029 function blocks = stk_parallel_cutblocks(x, ncores, min_block_size)
0030 
0031 x = double(x);
0032 n = size(x, 1);
0033 
0034 %--- Choose the actual block size & number of blocks ----------------------------
0035 
0036 nb_blocks = min(ncores, max(1, floor(n / min_block_size)));
0037 
0038 block_size = ceil(n / nb_blocks);
0039 
0040 %--- Cut blocks -----------------------------------------------------------------
0041 
0042 ind1 = 1 + ((1:nb_blocks)' - 1) * block_size;
0043 ind2 = min(n, ind1 + block_size - 1);
0044 
0045 % note: the last blocks can be slightly smaller than the others
0046 
0047 blocks = struct('i', cell(1, nb_blocks), 'j', [], 'xi', [], 'K', []);
0048 
0049 for b = 1:nb_blocks,
0050     
0051     blocks(b).i  = ind1(b):ind2(b);
0052     blocks(b).xi = x(blocks(b).i, :);
0053     
0054 end
0055 
0056 end % function stk_parallel_cutblocks
