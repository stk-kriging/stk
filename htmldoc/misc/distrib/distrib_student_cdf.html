0034 function [p, q] = distrib_student_cdf (x, nu, mu, sigma)
0035 
0036 if nargin > 4,
0037     stk_error ('Too many input arguments.', 'TooManyInputArgs');
0038 end
0039 
0040 if nargin > 2,
0041     x = bsxfun (@minus, x, mu);
0042 end
0043 
0044 if nargin > 3,
0045     x = bsxfun (@rdivide, x, sigma);
0046 end
0047 
0048 xx = x .^ 2;
0049 [x, xx, nu] = commonsize (x, xx, nu);
0050 
0051 % Return NaN for negative values of nu (or nu == NaN, or x == NaN)
0052 p = nan (size (x));
0053 q = nan (size (x));
0054 k0 = (nu > 0) & (~ isnan (x));
0055 
0056 % Gaussian case (nu = +inf)
0057 k_inf = isinf (nu);  k = k0 & k_inf;
0058 [p(k), q(k)] = distrib_normal_cdf (x(k));
0059 
0060 k0 = k0 & (~ k_inf);
0061 kp = (x > 0);
0062 kn = k0 & (~ kp);
0063 kp = k0 & kp;
0064 k_big_abs = (xx > nu);
0065 
0066 % Student case (nu < +inf) for positive x: compute q first, then p = 1 - q
0067 k = kp & k_big_abs;
0068 q(k) = betainc (nu(k) ./ (nu(k) + xx(k)), nu(k)/2, 1/2) / 2;
0069 k = kp & (~ k_big_abs);
0070 q(k) = 0.5 * (1 - betainc (xx(k) ./ (nu(k) + xx(k)), 1/2, nu(k)/2));
0071 p(kp) = 1 - q(kp);
0072 
0073 % Student case (nu < +inf) for negative x: compute p first, then q = 1 - p
0074 k = kn & k_big_abs;
0075 p(k) = betainc (nu(k) ./ (nu(k) + xx(k)), nu(k)/2, 1/2) / 2;
0076 k = kn & (~ k_big_abs);
0077 p(k) = 0.5 * (1 - betainc (xx(k) ./ (nu(k) + xx(k)), 1/2, nu(k)/2));
0078 q(kn) = 1 - p(kn);
0079 
0080 end % function distrib_student_cdf
