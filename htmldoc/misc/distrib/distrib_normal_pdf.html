0029 function density = distrib_normal_pdf (x, mu, sigma)
0030 
0031 if nargin > 3,
0032     stk_error ('Too many input arguments.', 'TooManyInputArgs');
0033 end
0034 
0035 if nargin > 1,
0036     x = bsxfun (@minus, x, mu);
0037 end
0038 
0039 if nargin > 2,
0040     [x, sigma] = commonsize (x, sigma);
0041     x = x ./ sigma;
0042     k0 = (sigma > 0);
0043 else
0044     sigma = 1;
0045     k0 = 1;
0046 end
0047 
0048 xx = x .^ 2;
0049 density = nan (size (x));
0050 
0051 k0 = k0 & (~ isnan (x));
0052 kb = (xx > 1491);  % when x^2 > 1491, the result is 0 in double precision
0053 
0054 % Deal with "large" values of abs(x)
0055 k = k0 & kb;
0056 density(k) = 0;
0057 
0058 % Deal with "small" values of abs(x)
0059 k = k0 & (~ kb);
0060 density(k) = 0.39894228040143268 * exp (- 0.5 * xx(k));
0061 
0062 density = bsxfun (@rdivide, density, sigma);
0063 
0064 end % function distrib_normal_pdf
