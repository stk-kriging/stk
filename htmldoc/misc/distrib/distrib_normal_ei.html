0030 function ei = distrib_normal_ei (x, mu, sigma, minimize)
0031 
0032 if nargin > 4,
0033     stk_error ('Too many input arguments.', 'TooManyInputArgs');
0034 end
0035 
0036 if nargin > 1,
0037     delta = bsxfun (@minus, mu, x);
0038 else
0039     % Default: mu = 0;
0040     delta = - x;
0041 end
0042 
0043 if nargin > 2,
0044     sigma(sigma < 0) = nan;
0045 else
0046     % Default: sigma = 1
0047     sigma = 1;
0048 end
0049 
0050 % Default: compute the EI for a maximization problem
0051 if nargin > 3,
0052     minimize = logical (minimize);
0053     if ~ minimize,
0054         delta = - delta;
0055     end
0056 end
0057 
0058 [delta, sigma] = commonsize (delta, sigma);
0059 
0060 ei = nan (size (delta));
0061 
0062 b0 = ~ (isnan (delta) | isnan (sigma));
0063 b1 = (sigma > 0);
0064 
0065 % Compute the EI where sigma > 0
0066 b = b0 & b1;
0067 u = delta(b) ./ sigma(b);
0068 ei(b) = sigma(b) .* (distrib_normal_pdf (u) + u .* distrib_normal_cdf (u));
0069 
0070 % Compute the EI where sigma == 0
0071 b = b0 & (~ b1);
0072 ei(b) = max (0, delta(b));
0073 
0074 % Correct numerical inaccuracies
0075 ei(ei < 0) = 0;
0076 
0077 end % function distrib_normal_ei
