0029 function [p, q] = distrib_normal_cdf (x, mu, sigma)
0030 
0031 if nargin > 3,
0032     stk_error ('Too many input arguments.', 'TooManyInputArgs');
0033 end
0034 
0035 if nargin > 1,
0036     x = bsxfun (@minus, x, mu);
0037 end
0038 
0039 if nargin > 2,
0040     [x, sigma] = commonsize (x, sigma);
0041     x = x ./ sigma;
0042     k0 = (sigma > 0);
0043 else
0044     k0 = 1;
0045 end
0046    
0047 p = nan (size (x));
0048 q = nan (size (x));
0049 
0050 k0 = k0 & (~ isnan (x));
0051 kp = (x > 0);
0052 kn = k0 & (~ kp);
0053 kp = k0 & kp;
0054 
0055 % Deal with positive values of x: compute q first, then p = 1 - q
0056 q(kp) = 0.5 * erfc (0.707106781186547524 * x(kp));
0057 p(kp) = 1 - q(kp);
0058 
0059 % Deal with negative values of x: compute p first, then q = 1 - p
0060 p(kn) = 0.5 * erfc (- 0.707106781186547524 * x(kn));
0061 q(kn) = 1 - p(kn);
0062 
0063 end % function distrib_normal_cdf
