0030 function ei = distrib_student_ei (x, nu, mu, sigma, minimize)
0031 
0032 if nargin > 5,
0033     stk_error ('Too many input arguments.', 'TooManyInputArgs');
0034 end
0035 
0036 nu(nu < 0) = nan;
0037 
0038 if nargin > 2,
0039     delta = bsxfun (@minus, mu, x);
0040 else
0041     % Default: mu = 0;
0042     delta = - x;
0043 end
0044 
0045 if nargin > 3,
0046     sigma(sigma < 0) = nan;
0047 else
0048     % Default
0049     sigma = 1;
0050 end
0051 
0052 % Default: compute the EI for a maximization problem
0053 if nargin > 4,    
0054     minimize = logical (minimize);
0055     if ~ minimize,
0056         delta = - delta;
0057     end
0058 end
0059 
0060 [delta, nu, sigma] = commonsize (delta, nu, sigma);
0061 
0062 ei = nan (size (delta));
0063 
0064 b0 = ~ (isnan (delta) | isnan (nu) | isnan (sigma));
0065 b1 = (nu > 1);   
0066 b2 = (sigma > 0);
0067 
0068 % The EI is infinite for nu <= 1
0069 ei(b0 & (~ b1)) = +inf;
0070 b0 = b0 & b1;
0071 
0072 % Compute the EI where nu > 1 and sigma > 0
0073 b = b0 & b2;  u = delta(b) ./ sigma(b);  nu = nu(b);
0074 ei(b) = sigma(b) .* ((nu + u .^ 2) ./ (nu - 1) ...
0075     .* distrib_student_pdf (u, nu) + u .* distrib_student_cdf (u, nu));
0076 
0077 % Compute the EI where nu > 1 and sigma == 0
0078 b = b0 & (~ b2);
0079 ei(b) = max (0, delta(b));
0080 
0081 % Correct numerical inaccuracies
0082 ei(ei < 0) = 0;
0083 
0084 end % function distrib_student_ei
