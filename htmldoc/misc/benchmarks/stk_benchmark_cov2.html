0029 %% Benchmark parameters
0030 
0031 DIM  = 1;
0032 
0033 N_START = 50;
0034 N_MAX   = 2000;
0035 
0036 covnames = { ...
0037     'stk_materncov32_iso', ...
0038     'stk_materncov52_iso', ...
0039     'stk_materncov_iso' };
0040 
0041 result = struct('covname', [], 't', [], 'n', []);
0042 
0043 
0044 %% Main loop
0045 
0046 for j = 1:4,
0047     
0048     switch j
0049         case 1
0050             disp('Using stk_parallel_engine_none...');
0051             stk_parallel_stop();
0052         case 2
0053             disp('parallelization on (if available) / msfb = Inf');
0054             stk_parallel_start();
0055             M = stk_options_get('stk_sf_matern', 'min_size_for_parallelization');
0056             stk_options_set('stk_sf_matern', 'min_size_for_parallelization', +Inf);
0057         case 3
0058             disp('parallelization on (if available) / msfb = 1');
0059             stk_options_set('stk_sf_matern', 'min_size_for_parallelization', 1);
0060         case 4
0061             fprintf('parallelization on (if available) / msfb = %d (default)', M);
0062             stk_options_set('stk_sf_matern', 'min_size_for_parallelization', M);
0063     end
0064     
0065     for k = 1:length(covnames), % loop over covariance functions
0066         
0067         result(j, k).covname = covnames{k};
0068         model = stk_model(covnames{k}, DIM);
0069         
0070         REP = 5000;
0071         n   = N_START;
0072         
0073         while n <= N_MAX
0074                        
0075             tic;
0076             x = stk_sampling_regulargrid(n, DIM);
0077             for i = 1:REP,
0078                 K = stk_make_matcov(model, x, x);
0079             end
0080             t = toc / REP;
0081             
0082             result(j, k).n(end+1) = n;
0083             result(j, k).t(end+1) = t;
0084             
0085             figure(1); cla;
0086             loglog(result(j, k).n, result(j, k).t, 'o-'); drawnow;
0087             
0088             n = ceil(n * 1.4);
0089             REP = ceil(1/t);
0090         end
0091     end
0092 end
0093
