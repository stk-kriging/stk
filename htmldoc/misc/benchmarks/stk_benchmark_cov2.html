0029 %% Benchmark parameters
0030 
0031 DIM  = 1;
0032 
0033 N_START = 50;
0034 N_MAX   = 2000;
0035 
0036 covnames = { ...
0037     'stk_materncov32_iso', ...
0038     'stk_materncov52_iso', ...
0039     'stk_materncov_iso' };
0040 
0041 result = struct('covname', [], 't', [], 'n', []);
0042 
0043 stk_figure ('stk_benchmark_cov2');
0044 
0045 
0046 %% Main loop
0047 
0048 for j = 1:4,
0049     
0050     switch j
0051         case 1
0052             disp('Using stk_parallel_engine_none...');
0053             stk_parallel_stop();
0054         case 2
0055             disp('parallelization on (if available) / msfb = Inf');
0056             stk_parallel_start();
0057             M = stk_options_get('stk_sf_matern', 'min_size_for_parallelization');
0058             stk_options_set('stk_sf_matern', 'min_size_for_parallelization', +Inf);
0059         case 3
0060             disp('parallelization on (if available) / msfb = 1');
0061             stk_options_set('stk_sf_matern', 'min_size_for_parallelization', 1);
0062         case 4
0063             fprintf('parallelization on (if available) / msfb = %d (default)', M);
0064             stk_options_set('stk_sf_matern', 'min_size_for_parallelization', M);
0065     end
0066     
0067     for k = 1:length(covnames), % loop over covariance functions
0068         
0069         result(j, k).covname = covnames{k};
0070         model = stk_model(covnames{k}, DIM);
0071         
0072         REP = 5000;
0073         n   = N_START;
0074         
0075         while n <= N_MAX
0076             
0077             tic;
0078             x = stk_sampling_regulargrid(n, DIM);
0079             for i = 1:REP,
0080                 K = stk_make_matcov(model, x, x);
0081             end
0082             t = toc / REP;
0083             
0084             result(j, k).n(end+1) = n;
0085             result(j, k).t(end+1) = t;
0086             
0087             cla;  loglog(result(j, k).n, result(j, k).t, 'o-');  drawnow;
0088             
0089             n = ceil(n * 1.4);
0090             REP = ceil(1/t);
0091         end
0092     end
0093 end
0094 
0095
