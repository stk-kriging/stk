0029 function [str, err] = stk_sprintf_colvect_scientific(x, max_width)
0030 
0031 if isempty(x),
0032     str = '';
0033     err = 0;
0034     return;
0035 end
0036 
0037 if nargin < 2,
0038     max_width = 10;
0039 end
0040 
0041 % turn x into a column vector
0042 x = double(x);
0043 x = x(:);
0044 
0045 % get rid of negative zeros
0046 b = (x == 0); 
0047 x(b) = +0.0;
0048 
0049 % compute exponents
0050 ax = abs(x);
0051 exponent = zeros(size(x));
0052 exponent(~b) = floor(log10(ax(~b)));
0053 
0054 % compute mantissae
0055 mantissa = x .* 10.^(-exponent);
0056 
0057 % maximal absolute exponent
0058 maxexp = max(abs(exponent));
0059 
0060 % is there any negative element ?
0061 any_negative = any(x < 0);
0062 
0063 % Start with only one digit for the mantissa (no comma, then)
0064 n1 = 1;                                 % nb digits for the mantissa
0065 n2 = max(2, 1 + floor(log10(maxexp)));  % nb digits for the exponent
0066 n3 = any_negative + 2;                  % nb non-digit characters (+2 for "e+")
0067 
0068 % Abort this is already too long
0069 if (n1 + n2 + n3) > max_width
0070     str = repmat('#', length(x), max_width);
0071     err = +inf;
0072     return;
0073 end
0074 
0075 % Otherwise, this is our current best solution
0076 am = abs(mantissa);
0077 best_err = max(am - floor(am)); % maximal error on the matissa
0078 best_n1  = 1;
0079 
0080 % Should we add a decimal part ?
0081 if (best_err > eps) && ((n1 + n2 + n3 + 2) <= max_width)
0082     % We can add a decimal part, so let's do it...
0083     while (best_err > eps) && ((n1 + n2 + n3) < max_width)
0084         n1 = n1 + 1;
0085         n3 = any_negative + 3;
0086         % "+3" for "e", "." in the mantissa and "+" in the exponent
0087         c = 10^(1 - n1);
0088         mm = round(am ./ c) .* c;
0089         err = max(abs(am - mm));
0090         if err < best_err - 0.5 * c
0091             best_err = err;
0092             best_n1 = n1;
0093         end
0094     end
0095 end
0096 n1 = best_n1;
0097 
0098 % format specifier for the mantissa
0099 fmt1 = sprintf('%%%d.%df', n1 + (n1 > 1) + any_negative, n1 - 1);
0100 str1 = arrayfun(@(u)(sprintf(fmt1, u)), mantissa, 'UniformOutput', false);
0101 
0102 % format specifier for the exponent
0103 fmt2 = sprintf('e%%+0%dd', n2 + 1);
0104 str2 = arrayfun(@(u)(sprintf(fmt2, u)), exponent, 'UniformOutput', false);
0105 
0106 str = [char(str1{:}) char(str2{:})];
0107 
0108 % Compute the maximal error
0109 if nargout > 1,
0110     c = 10^(1 - n1);
0111     rounded_mantissa = round(mantissa / c) * c;
0112     err_mantissa = abs(rounded_mantissa - mantissa);
0113     err = max(err_mantissa .* 10.^(exponent));
0114 end
