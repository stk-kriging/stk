0030 function s = stk_sprintf_data (x, max_width, colnames, rownames)
0031 
0032 if ~ isnumeric (x)
0033     
0034     errmsg = sprintf ('Incorrect argument type: %s', class (x));
0035     stk_error (errmsg, 'IncorrectType');
0036     
0037 else
0038     
0039     x = double (x);
0040     [n, d] = size (x);
0041     
0042     if (n == 0) || (d == 0)
0043         
0044         s = '[] (empty)';
0045         
0046     else
0047         
0048         if nargin < 3,
0049             colnames = {};
0050         end
0051         
0052         if nargin < 4,
0053             rownames = {};
0054         end
0055         
0056         if (nargin < 2) || isempty (max_width)
0057             try
0058                 switch get (0, 'Format')
0059                     case 'short'
0060                         max_width = 6;
0061                     case 'long'
0062                         max_width = 16;
0063                     otherwise
0064                         % FIXME: handle other formatting modes...
0065                         max_width = 10;
0066                 end
0067             catch
0068                 % Property 'Format' doesn't exist in Octave 3.2.x
0069                 max_width = 6;
0070             end
0071         end
0072         
0073         nb_spaces_colsep = 2;
0074         
0075         has_colnames = ~ isempty (colnames);
0076         nb_rows = n + has_colnames;
0077         
0078         s = repmat ('', nb_rows, 1); %#ok<*AGROW>
0079         
0080         % first columns: row names
0081         if isempty (rownames)
0082             rownames = stk_sprintf_colvect (1:n);
0083         else
0084             rownames = char (rownames);
0085         end
0086         
0087         if has_colnames
0088             s = [s [repmat(' ', 1, size(rownames, 2)); rownames]];
0089         else
0090             s = [s rownames];
0091         end
0092         
0093         % column separator between row names and the first data column
0094         s = [s repmat(' :', nb_rows,  1) ...
0095             repmat(' ', nb_rows,  nb_spaces_colsep)];
0096         
0097         for j = 1:d,
0098             
0099             xx  = stk_sprintf_colvect (x(:, j), max_width);
0100             
0101             if ~ has_colnames % no need for a header row in this case
0102                 
0103                 s = [s xx];
0104                 
0105             else
0106                 
0107                 vn = colnames{j};
0108                 Lxx = size (xx, 2);
0109                 L = max (length (vn), Lxx);
0110                 s = [s [sprintf('% *s', L, vn);  % variable name
0111                     repmat(' ', n, L - Lxx) xx]];    % formatted data
0112                 
0113             end
0114             
0115             if j < d,
0116                 % column separator
0117                 s = [s repmat(' ', nb_rows,  nb_spaces_colsep)];
0118             end
0119             
0120         end % for
0121         
0122     end % if
0123     
0124 end % function stk_sprintf_data
