0029 function [str, err] = stk_sprintf_colvect_fixedpoint(x, max_width)
0030 
0031 if isempty(x),
0032     str = '';
0033     err = 0;
0034     return;
0035 end
0036 
0037 if nargin < 2,
0038     max_width = 10;
0039 end
0040 
0041 % turn x into a column vector
0042 x = double(x);
0043 x = x(:);
0044 
0045 % get rid of negative zeros
0046 b = (x == 0);
0047 x(b) = +0.0;
0048 
0049 % Is there any negative element ?
0050 any_negative = any(x < 0);
0051 
0052 % Start without decimal part
0053 ax = abs(x);
0054 n1 = max(1, floor(log10(max(ax))) + 1);
0055 n2 = 0;
0056 n3 = any_negative;
0057 
0058 % Abort this is already too long
0059 if (n1 + n2 + n3) > max_width
0060     str = repmat('#', length(x), max_width);
0061     err = +inf;
0062     return;
0063 end
0064 
0065 % Otherwise, this is our current best solution
0066 best_err = max(abs(fix(x) - x));
0067 best_n2  = 0;
0068 
0069 % Should we add a decimal part ?
0070 if (best_err > eps) && ((n1 + n2 + n3 + 2) <= max_width)
0071     % We can add a decimal part, so let's do it...
0072     while (best_err > eps) && ((n1 + n2 + n3) < max_width)
0073         n2  = n2 + 1; % add one decimal
0074         n3  = 1 + any_negative; % +1 for the comma
0075         c   = 10^(-n2);
0076         xx  = floor(ax / c) * c;
0077         err = max(ax - xx);
0078         if err < best_err - 0.5*c,
0079             best_err = err;
0080             best_n2  = n2;
0081         end
0082     end
0083 end
0084 err = best_err;
0085 n2  = best_n2;
0086 
0087 fmt = sprintf('%%%d.%df', n1 + n2 + n3, n2);
0088 str = arrayfun(@(u)(sprintf(fmt, u)), x, 'UniformOutput', false);
0089 str = char(str{:});
0090 
0091 end % function stk_sprintf_colvect_fixedpoint
