0030 function stk_error(errmsg, mnemonic, stack)
0031 
0032 % second component of error identifier = name of calling function
0033 % (unless stk_error has been called directly from the base workspace)
0034 if nargin < 3,
0035     stack = dbstack();
0036     if length(stack) == 1,
0037         caller = 'BaseWorkspace';
0038     else
0039         % pretend that the error has been thrown by the caller
0040         stack = stack(2:end);
0041         % and get caller name to form the error identifier
0042         caller = stack(1).name;
0043     end
0044 else
0045     % if a "stack" has been provided by the user, we use it without asking
0046     % questions (if it's a struct)
0047     if ~isstruct(stack) || ~isfield(stack, 'name'),
0048         % We will throw an error, but not the one we were requested to!
0049         errmsg = 'Argument "stack" should be a valid stack structure.';
0050         mnemonic = 'InvalidArgument';
0051         stack = dbstack();
0052     end
0053     caller = stack(1).name;
0054 end
0055 
0056 % keep only subfunction name (Octave)
0057 gt_pos = strfind(caller, '>');
0058 if ~isempty(gt_pos),
0059     caller = caller((gt_pos + 1):end);
0060 end
0061 
0062 % construct the error structure
0063 errstruct = struct( ...
0064     'message', errmsg, ...
0065     'identifier', sprintf('STK:%s:%s', caller, mnemonic), ...
0066     'stack', stack);
0067 
0068 error(errstruct);
0069 
0070 end % stk_error
