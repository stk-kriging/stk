0042 % Check that box is a numeric matrix (the behaviour of ismatrix is inconsistent
0043 % between Matlab and Octave for arrays with more than two dimensions...)
0044 ismatrix = isnumeric(box) && (length(size(box)) == 2);
0045 if ~ismatrix,
0046     errmsg = sprintf('%s is not a numeric matrix.', boxname);
0047     stk_error(errmsg, mnemonic, stack);
0048 end
0049 
0050 [n, d] = size(box);
0051 if n ~= 2,
0052     errmsg = sprintf('%s should have two rows, not more, not less.', boxname);
0053     stk_error(errmsg, mnemonic, stack);
0054 end
0055 
0056 if d == 0,
0057     errmsg = sprintf('%s should have at least one column.', boxname);
0058     stk_error(errmsg, mnemonic, stack);
0059 end
0060 
0061 if (nargin == 2) && (d ~= dim)
0062     errmsg = sprintf('%s should exactly %d columns.', boxname, dim);
0063     stk_error(errmsg, mnemonic, stack);
0064 end
0065 
0066 if any(isnan(box)),
0067     errmsg = sprintf('%s should at least one column.', boxname);
0068     stk_error(errmsg, mnemonic, stack);
0069 end
0070 
0071 if any(box(1, :) > box(2, :)),
0072     errmsg = sprintf('%s has invalid boundaries.', boxname);
