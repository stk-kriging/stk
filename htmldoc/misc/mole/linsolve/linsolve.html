0017 ## @deftypefn  {Function File} {@var{x} =} linsolve (@var{A}, @var{b})
0018 ## @deftypefnx {Function File} {@var{x} =} linsolve (@var{A}, @var{b}, @var{opts})
0019 ## @deftypefnx {Function File} {[@var{x}, @var{R}] =} linsolve (@dots{})
0020 ## Solve the linear system @code{A*x = b}.
0021 ##
0022 ## With no options, this function is equivalent to the left division operator
0023 ## @w{(@code{x = A \ b})} or the matrix-left-divide function
0024 ## @w{(@code{x = mldivide (A, b)})}.
0025 ##
0026 ## Octave ordinarily examines the properties of the matrix @var{A} and chooses
0027 ## a solver that best matches the matrix.  By passing a structure @var{opts}
0028 ## to @code{linsolve} you can inform Octave directly about the matrix @var{A}.
0029 ## In this case Octave will skip the matrix examination and proceed directly
0030 ## to solving the linear system.
0031 ##
0032 ## @strong{Warning:} If the matrix @var{A} does not have the properties
0033 ## listed in the @var{opts} structure then the result will not be accurate
0034 ## AND no warning will be given.  When in doubt, let Octave examine the matrix
0035 ## and choose the appropriate solver as this step takes little time and the
0036 ## result is cached so that it is only done once per linear system.
0037 ##
0038 ## Possible @var{opts} fields (set value to true/false):
0039 ##
0040 ## @table @asis
0041 ## @item LT
0042 ##   @var{A} is lower triangular
0043 ##
0044 ## @item UT
0045 ##   @var{A} is upper triangular
0046 ##
0047 ## @item UHESS
0048 ##   @var{A} is upper Hessenberg (currently makes no difference)
0049 ##
0050 ## @item SYM
0051 ##   @var{A} is symmetric or complex Hermitian (currently makes no difference)
0052 ##
0053 ## @item POSDEF
0054 ##   @var{A} is positive definite
0055 ##
0056 ## @item RECT
0057 ##   @var{A} is general rectangular (currently makes no difference)
0058 ##
0059 ## @item TRANSA
0060 ##   Solve @code{A'*x = b} by @code{transpose (A) \ b}
0061 ## @end table
0062 ##
0063 ## The optional second output @var{R} is the inverse condition number of
0064 ## @var{A} (zero if matrix is singular).
0065 ## @seealso{mldivide, matrix_type, rcond}
0066 ## @end deftypefn

0076 function [x, R] = linsolve (A, b, opts)
0077 
0078   if (nargin < 2 || nargin > 3)
0079     print_usage ();
0080   endif
0081 
0082   if (! (isnumeric (A) && isnumeric (b)))
0083     error ("linsolve: A and B must be numeric");
0084   endif
0085 
0086   ## Process any opts
0087   if (nargin > 2)
0088     if (! isstruct (opts))
0089       error ("linsolve: OPTS must be a structure");
0090     endif
0091     trans_A = false;
0092     if (isfield (opts, "TRANSA") && opts.TRANSA)
0093       trans_A = true;
0094       A = A';
0095     endif
0096     if (isfield (opts, "POSDEF") && opts.POSDEF)
0097       A = matrix_type (A, "positive definite");
0098     endif  
0099     if (isfield (opts, "LT") && opts.LT)
0100       if (trans_A)
0101         A = matrix_type (A, "upper");
0102       else
0103         A = matrix_type (A, "lower");
0104       endif
0105     endif
0106     if (isfield (opts, "UT") && opts.UT)
0107       if (trans_A)
0108         A = matrix_type (A, "lower");
0109       else
0110         A = matrix_type (A, "upper");
0111       endif
0112     endif        
0113   endif
0114 
0115   x = A \ b;
0116 
0117   if (nargout > 1)
0118     if (issquare (A))
0119       R = rcond (A);
0120     else
0121       R = 0;
0122     endif
0123   endif
0124 endfunction # linsolve
