0050 function c = corr (x, y)
0051 
0052 % Note: the original Octave code has been rewritten to avoid calling cov, since
0053 % there is a long-standing incompatiblity between Matlab's cov and Octave's cov
0054 % (see https://savannah.gnu.org/bugs/?40751)
0055 
0056 n = size (x, 1);
0057 
0058 x = x - repmat (mean (x), n, 1);
0059 sx = std (x, 1);
0060 sx (sx == 0) = nan;
0061 
0062 if nargin < 2,
0063     y = x;
0064     sy = sx;
0065 else
0066     assert (size (y, 1) == n);
0067     y = y - repmat (mean (y), n, 1);
0068     sy = std (y, 1)
0069     sy (sy == 0) = nan;
0070 end
0071 
0072 % Special case, scalar is always 100% correlated with itself
0073 if isscalar (x)
0074     if isa (x, 'single') && isa (y, 'single')
0075         c = single (1.0);
0076     else
0077         c = 1.0;
0078     end
0079     return;
0080 end
0081 
0082 c = x' * y;
0083 s = sx' * sy;
0084 c = c ./ (n * s);
0085 
0086 end % function corr
