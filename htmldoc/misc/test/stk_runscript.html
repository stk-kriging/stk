0029 function err = stk_runscript(scriptname)
0030 
0031 has_dot_m = strcmp(scriptname(end-1:end), '.m');
0032 
0033 if stk_is_octave_in_use(),
0034     % Octave's run() wants scriptnames WITH a .m in 3.6.2
0035     % (both forms were accepted in 3.2.4... damn it...)
0036     if ~has_dot_m,
0037         scriptname = [scriptname '.m'];
0038     end
0039 else
0040     % Matlab wants scriptnames WITHOUT a .m
0041     if has_dot_m,
0042         scriptname = scriptname(1:end-2);
0043     end
0044 end
0045 
0046 err = [];
0047 
0048 try
0049     run(scriptname);
0050 catch %#ok<CTCH>
0051     err = lasterror(); %#ok<LERR>
0052 end
0053 
0054 end % function stk_runscript
