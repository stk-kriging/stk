0048 function stk_runtests(directory)
0049 
0050 if nargin == 0
0051     % scan all STK directories if no input argument is provided
0052     dirs = stk_path();
0053 else
0054     % otherwise, directory is expected to be a valid directory name
0055     if ~exist(directory, 'dir')
0056         error('Directory not found.');
0057     end
0058     here = pwd();
0059     cd(directory);
0060     dirs = {pwd()}; % get absolute path
0061     cd(here);
0062 end
0063 
0064 % number of directories to be explored
0065 nb_topdirs = numel(dirs);
0066 
0067 % run tests all available tests in each directory
0068 n_total = 0; n_pass = 0; n_files = 0; n_notest = 0; n_dirs = 0;
0069 for i = 1:nb_topdirs
0070     [np, nt, nn, nf, nd] = run_all_tests(dirs{i}, dirs{i});
0071     n_total  = n_total  + nt;
0072     n_pass   = n_pass   + np;
0073     n_files  = n_files  + nf;
0074     n_notest = n_notest + nn;
0075     n_dirs   = n_dirs   + nd;
0076 end
0077 
0078 if n_dirs > 1,
0079     fprintf('Summary for all %d directories:\n', n_dirs);
0080     fprintf(' --> passed %d/%d tests\n', n_pass, n_total);
0081     fprintf(' --> %d/%d files had no tests\n', n_notest, n_files);
0082 end
0083 
0084 end

0091 function [n_pass, n_total, n_notest, n_files, n_dirs] = run_all_tests(testdir, basedir)
0092 
0093 % list directory content
0094 dirinfo = dir(testdir);
0095 flist = {dirinfo.name};
0096 
0097 here = pwd(); cd(basedir);
0098 
0099 fprintf ('Processing files in %s:\n\n', testdir);
0100 fflush (stdout);
0101 
0102 % init counters
0103 n_total  = 0;
0104 n_pass   = 0;
0105 n_files  = 0;
0106 n_notest = 0;
0107 n_dirs   = 1;
0108 
0109 % list of subdirectories to be processed
0110 subdirs_class = {};
0111 subdirs_private = {};
0112 
0113 for i = 1:numel (flist)
0114     f = flist{i};
0115     ff = fullfile(testdir, f);
0116     if (length (f) > 2) && strcmp (f((end-1):end), '.m')
0117         n_files = n_files + 1;
0118         print_test_file_name (f);
0119         if has_tests(ff)
0120             % Silence all warnings & prepare for warning detection.
0121             s = warning('off', 'all'); lastwarn('');
0122             % Do the actual tests.
0123             [p, n] = stk_test (ff, 'quiet', stdout);
0124             % Note: the presence of the third argument (fid=stdout) forces
0125             % stk_test in batch mode, which means that it doesn't stop at
0126             % the first failed test.
0127             print_pass_fail(n, p);
0128             n_total = n_total + n;
0129             n_pass  = n_pass  + p;
0130             % deal with warnings
0131             if ~isempty(lastwarn()), fprintf(' (warnings)'); end
0132             warning(s);
0133         else
0134             n_notest = n_notest + 1;
0135             fprintf(' NO TESTS');
0136         end
0137         fprintf('\n');
0138         fflush(stdout);
0139     elseif dirinfo(i).isdir && (f(1) == '@')
0140         subdirs_class{end+1} = ff; %#ok<AGROW>

0176 function retval = has_tests (f)
0177 
0178 fid = fopen (f);
0179 if (fid >= 0)
