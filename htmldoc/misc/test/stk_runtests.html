0048 function stk_runtests(directory)
0049 
0050 if nargin == 0
0051     % scan all STK directories if no input argument is provided
0052     dirs = stk_path();
0053 else
0054     % otherwise, directory is expected to be a valid directory name
0055     if ~exist(directory, 'dir')
0056         error('Directory not found.');
0057     end
0058     here = pwd();
0059     cd(directory);
0060     dirs = {pwd()}; % get absolute path
0061     cd(here);
0062 end
0063 
0064 % number of directories to be explored
0065 nb_dirs = numel(dirs);
0066 
0067 % run tests all available tests in each directory
0068 n_total = 0; n_pass = 0; n_files = 0; n_notest = 0;
0069 for i = 1:nb_dirs
0070     [n_pass_, n_total_, n_notest_, n_files_] = run_all_tests(dirs{i}, dirs{i});
0071     n_total  = n_total + n_total_;
0072     n_pass   = n_pass + n_pass_;
0073     n_files  = n_files + n_files_;
0074     n_notest = n_notest + n_notest_;
0075 end
0076 
0077 if nb_dirs > 1,
0078     fprintf('Summary for all %d directories:\n', nb_dirs);
0079     fprintf(' --> passed %d/%d tests\n', n_pass, n_total);
0080     fprintf(' --> %d/%d files had no tests\n', n_notest, n_files);
0081 end
0082 
0083 end

0090 function [n_pass, n_total, n_notest, n_files] = run_all_tests(testdir, basedir)
0091 
0092 % list directory content
0093 dirinfo = dir(testdir);
0094 flist = {dirinfo.name};
0095 
0096 here = pwd(); cd(basedir);
0097 
0098 fprintf ('Processing files in %s:\n\n', testdir);
0099 fflush (stdout);
0100 
0101 % init counters
0102 n_total  = 0;
0103 n_pass   = 0;
0104 n_files  = 0;
0105 n_notest = 0;
0106 
0107 % list of subdirectories to be processed
0108 subdirs_class = {};
0109 subdirs_private = {};
0110 
0111 for i = 1:numel (flist)
0112     f = flist{i};
0113     ff = fullfile(testdir, f);
0114     if (length (f) > 2) && strcmp (f((end-1):end), '.m')
0115         n_files = n_files + 1;
0116         print_test_file_name (f);
0117         if has_tests(ff)
0118             [p, n] = stk_test (ff, 'quiet', stdout);
0119             % Note: the presence of the third argument (fid=stdout) forces
0120             % stk_test in batch mode, which means that it doesn't stop at
0121             % the first failed test.
0122             print_pass_fail(n, p);
0123             n_total = n_total + n;
0124             n_pass  = n_pass  + p;
0125         else
0126             n_notest = n_notest + 1;
0127             fprintf(' NO TESTS\n');
0128         end
0129         fflush(stdout);
0130     elseif dirinfo(i).isdir && (f(1) == '@')
0131         subdirs_class{end+1} = ff; %#ok<AGROW>
0132     elseif dirinfo(i).isdir && strcmp(f, 'private')
0133         subdirs_private{end+1} = ff; %#ok<AGROW>
0134     end
0135 end
0136 fprintf('   --> passed %d/%d tests\n', n_pass, n_total);
0137 fprintf('   --> %d/%d files had no tests\n', n_notest, n_files);
0138 fprintf('\n');
0139 
0140 for i = 1:length(subdirs_class)
0141     [p, n, nnt, nf] = run_all_tests(subdirs_class{i}, pwd());
0142     n_total  = n_total  + n;
0143     n_pass   = n_pass   + p;

0145     n_notest = n_notest + nnt;

0165 function retval = has_tests (f)
0166

0183 function print_pass_fail (n, p)
0184
