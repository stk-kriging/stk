0048 function stk_runtests (directory)
0049 
0050 if nargin == 0
0051     % scan all STK directories if no input argument is provided
0052     dirs = stk_path();
0053 else
0054     % otherwise, directory is expected to be a valid directory name
0055     if ~exist(directory, 'dir')
0056         error('Directory not found.');
0057     end
0058     here = pwd();
0059     cd(directory);
0060     dirs = {pwd()}; % get absolute path
0061     cd(here);
0062 end
0063 
0064 % number of directories to be explored
0065 nb_topdirs = numel(dirs);
0066 
0067 % run tests all available tests in each directory
0068 n_total = 0; n_pass = 0; n_files = 0; n_notest = 0; n_dirs = 0;
0069 for i = 1:nb_topdirs
0070     [np, nt, nn, nf, nd] = run_all_tests(dirs{i}, dirs{i});
0071     n_total  = n_total  + nt;
0072     n_pass   = n_pass   + np;
0073     n_files  = n_files  + nf;
0074     n_notest = n_notest + nn;
0075     n_dirs   = n_dirs   + nd;
0076 end
0077 
0078 if n_dirs > 1,
0079     fprintf('Summary for all %d directories:\n', n_dirs);
0080     fprintf(' --> passed %d/%d tests\n', n_pass, n_total);
0081     fprintf(' --> %d/%d files had no tests\n', n_notest, n_files);
0082 end
0083 
0084 end % function stk_runtests

0091 function [n_pass, n_total, n_notest, n_files, n_dirs] ...
0092     = run_all_tests (testdir, basedir)
0093 
0094 % list directory content
0095 dirinfo = dir(testdir);
0096 flist = {dirinfo.name};
0097 
0098 here = pwd(); cd(basedir);
0099 
0100 fprintf ('Processing files in %s:\n\n', testdir);
0101 fflush (stdout);
0102 
0103 % init counters
0104 n_total  = 0;
0105 n_pass   = 0;
0106 n_files  = 0;
0107 n_notest = 0;
0108 n_dirs   = 1;
0109 
0110 % list of subdirectories to be processed
0111 subdirs_class = {};
0112 subdirs_private = {};
0113 
0114 for i = 1:numel (flist)
0115     f = flist{i};
0116     ff = fullfile(testdir, f);
0117     if (length (f) > 2) && strcmp (f((end-1):end), '.m')
0118         n_files = n_files + 1;
0119         print_test_file_name (f);
0120         if has_tests(ff)
0121             % Silence all warnings & prepare for warning detection.
0122             s = warning('off', 'all'); lastwarn('');
0123             % Do the actual tests.
0124             [p, n] = stk_test (ff, 'quiet', stdout);
0125             % Note: the presence of the third argument (fid=stdout) forces
0126             % stk_test in batch mode, which means that it doesn't stop at
0127             % the first failed test.
0128             print_pass_fail(n, p);
0129             n_total = n_total + n;
0130             n_pass  = n_pass  + p;
0131             % deal with warnings
0132             if ~isempty(lastwarn()), fprintf(' (warnings)'); end
0133             warning(s);
0134         else
0135             n_notest = n_notest + 1;
0136             fprintf(' NO TESTS');
0137         end
0138         fprintf('\n');
0139         fflush(stdout);
0140     elseif dirinfo(i).isdir && (f(1) == '@')
0141         subdirs_class{end+1} = ff; %#ok<AGROW>
0142         n_dirs = n_dirs + 1;
0143     elseif dirinfo(i).isdir && strcmp(f, 'private')
0144         subdirs_private{end+1} = ff; %#ok<AGROW>
0145         n_dirs = n_dirs + 1;
0146     end

0177 function retval = has_tests (f)
0178 
0179 fid = fopen (f);
0180 if (fid >= 0)
0181     str = fread (fid, '*char')';
0182     fclose (fid);

0195 function print_pass_fail (n, p)
