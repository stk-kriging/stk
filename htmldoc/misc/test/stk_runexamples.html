0029 function stk_runexamples()
0030 
0031 % run all examples, recursively
0032 example_dir = fullfile(stk_get_root(), 'examples');
0033 [scriptname, err] = stk_runexamples_recurs_(example_dir);
0034 
0035 % display a summary
0036 fprintf('%s\n', stk_sprintf_framed('stk_runexamples summary'));
0037 
0038 for i = 1:length(scriptname)
0039     fprintf('[%02d] %s %s ', i, scriptname{i}, ...
0040         repmat('.', 1, 30 - length(scriptname{i})));
0041     if isempty(err{i})
0042         fprintf('OK\n');
0043     else
0044         id = err{i}.identifier;
0045         if isempty(id),
0046             fprintf('ERROR (no identifier provided)\n');
0047         else
0048             fprintf('%s\n', id);
0049         end
0050     end
0051 end
0052 
0053 assignin('base', 'stkRunExamplesErrors', err);
0054 
0055 if ~ all (cellfun (@isempty, err))
0056     error ('Some examples failed to run properly.');
0057 end
0058     
0059 fprintf('\n');
0060 
0061 end % function stk_runexamples

0064 function [scriptname, err] = stk_runexamples_recurs_(example_dir)
0065 
0066 s = dir(example_dir);
0067 
0068 scriptname = {};
0069 err = {};
0070 
0071 for i = 1:length(s),
0072     
0073     if s(i).isdir && (s(i).name(1) ~= '.')
0074         [n, e] = stk_runexamples_recurs_(fullfile(example_dir, s(i).name));
0075         scriptname = [scriptname n];
0076         err = [err e];
0077     else
0078         if ~isempty(regexp(s(i).name, '^stk_example.*.m$'))
0079             scriptname{end+1} = s(i).name(1:end-2);
0080             err{end+1} = stk_runscript(fullfile(example_dir, scriptname{end}));
0081             drawnow; pause(1.0); close all;
0082         end
0083     end
0084     
0085 end
0086 
0087 end % function stk_runexamples_recurs_
