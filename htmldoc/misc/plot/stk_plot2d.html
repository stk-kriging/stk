0033 function stk_plot2d(plotfun, x, arg3, varargin)
0034 
0035 
0036 %%% Check input arguments
0037 
0038 if ~isstruct(x) || ~isfield(x, 'coord') || ~isfield(x, 'a')
0039     errmsg = 'x should be a structure with fields ''a'' and ''coord''.';
0040     stk_error(errmsg, 'IncorrectArgument');
0041 end
0042 
0043 dim = size(x.a, 2);
0044 
0045 if dim ~= 2,
0046     errmsg = 'stk_plot2d only works for two-dimensional factor spaces.';
0047     stk_error(errmsg, 'IncorrectArgument');
0048 end
0049 
0050 if ~iscell(x.coord) || ~isequal(size(x.coord), [1 dim])
0051     errmsg = 'Field ''coord'' seems corrupted...';
0052     stk_error(errmsg, 'IncorrectArgument');
0053 end
0054 
0055 
0056 %%% Extract ndgrid-style coordinate matrices
0057 
0058 xx1 = x.coord{1};
0059 xx2 = x.coord{2};
0060 
0061 [n1, n2] = size(xx1);
0062 
0063 if ~isequal(size(xx2), [n1 n2])
0064     errmsg = 'Field ''coord'' seems corrupted...';
0065     stk_error(errmsg, 'IncorrectArgument');
0066 end
0067 
0068 
0069 %%% Deal with various possible types for the 'z' argument
0070 
0071 if ischar(arg3) || isa(arg3, 'function_handle')
0072     z = stk_feval(arg3, x);
0073     z = z.a;
0074 elseif isstruct(arg3),
0075     z = arg3.a;
0076 elseif isa(arg3, 'double')
0077     z = arg3;
0078 else
0079     stk_error('Unsupported type for the third argument', 'IncorrectArgument');
0080 end
0081 
0082 
0083 %%% Do the actual plotting job
0084 
0085 plotfun(xx1, xx2, reshape(z, n1, n2), varargin{:});
0086 
0087 xlabel('x_1', 'FontWeight', 'bold');
0088 ylabel('x_2', 'FontWeight', 'bold');
0089 
0090 if ismember(func2str(plotfun), {'surf', 'pcolor'}),
0091     shading('interp');
0092 end

0094 
0095 end % function stk_plot2d
