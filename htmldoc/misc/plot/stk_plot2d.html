0034 function stk_plot2d(plotfun, x, arg3, varargin)
0035 
0036 
0037 %%% Check input arguments
0038 
0039 if ~isa(x, 'stk_factorialdesign')
0040     errmsg = 'x should be an stk_factorialdesign object.';
0041     stk_error(errmsg, 'IncorrectArgument');
0042 end
0043 
0044 dim = size(x, 2);
0045 
0046 if dim ~= 2,
0047     errmsg = 'stk_plot2d only works for two-dimensional factor spaces.';
0048     stk_error(errmsg, 'IncorrectArgument');
0049 end
0050 
0051 
0052 %%% Deal with various possible types for the 'z' argument
0053 
0054 if ischar(arg3) || isa(arg3, 'function_handle')
0055     z = double(stk_feval(arg3, x));
0056 else
0057     z = double(arg3);
0058 end
0059 
0060 
0061 %%% Do the actual plotting job
0062 
0063 [xx1, xx2] = ndgrid(x);
0064 
0065 plotfun(xx1, xx2, reshape(z, size(xx1)), varargin{:});
0066 
0067 xlabel('x_1', 'FontWeight', 'bold');
0068 ylabel('x_2', 'FontWeight', 'bold');
0069 
0070 if ismember(func2str(plotfun), {'surf', 'pcolor'}),
0071     shading('interp');
