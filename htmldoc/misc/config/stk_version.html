0029 function v = stk_version ()
0030 
0031 persistent version_number
0032 
0033 if isempty (version_number)
0034     
0035     % open README
0036     filename = fullfile (stk_get_root (), 'README');
0037     fid = fopen (filename, 'rt');
0038     if fid == -1,
0039         stk_error ('Unable to open the README file.', 'FOpenFailed');
0040     end
0041     
0042     % what we're looking for
0043     t = 'Version: ';
0044     
0045     while 1
0046         
0047         % read a new line
0048         s = fgetl (fid);
0049         if s == -1,
0050             fclose (fid);
0051             stk_error ('Unable to read STK''s version number.', 'Unexpected');
0052         end
0053         
0054         % look for the pattern
0055         i = strfind (s, t);
0056         if ~isempty (i)
0057             version_number = strtrim (s((i + length(t)):end));
0058             fclose (fid);
0059             break
0060         end
0061         
0062     end
0063         
0064 end
0065
