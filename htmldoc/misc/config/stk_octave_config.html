0031 % We need to check that the GLPK library is installed. This is the case
0032 % in most recent releases of Octave, but some older releases do not contain
0033 % GLPK.

0035     stk_test_glpk_();
0036 catch %#ok<CTCH>
0037     error('The GLPK library does not seem to be properly installed');
0038 end

0040 % Note: simply checking that __glpk__.oct is present is not good enough,
0041 % since some distribution include (or so it seems) a fake __glpk__.oct
0042 % package, which is in charge of issuing an error message...
0043 
0044 % Suppress additional help information in Octave
0045 suppress_verbose_help_message(true);
0046 
0047 % Perhaps to early to use FLTK as a default choice
0048 % if ismember('fltk', available_graphics_toolkits())

0052 % So, we keep gnuplot as a graphical backend
0053 % but use GNUTERM=wxt for prettier and faster plots!

0055     % doesn't work in old Octave versions (e.g., 3.2.3)
0056     if ismember ('gnuplot', available_graphics_toolkits)
0057         graphics_toolkit ('gnuplot')
0058         setenv ('GNUTERM', 'wxt')
0059         fprintf ('Graphics toolkit: gnuplot\n');
0060     end
0061 end

0079 function stk_test_glpk_()
0080 
0081 % minimize c*x
0082 % under a*x = b, x >= 0
0083 a = 1;
0084 b = 1;
0085 c = 1;
0086 
0087 % solve this difficult problem using GLPK ;-)
0088 x = glpk (c, a, b);
0089 assert(x == 1);
0090 
0091 end
